99// Poisson likelihood calc for data and MC event rates
100double samplePDFBase::getTestStatLLH(const double data, const double mc) const {
101// ***************************************************************************
102 // Need some MC
103 if(mc == 0) return 0.;
104
105 double negLogL = 0;
106 if(mc > 0 && data > 0)
107 {
108 //http://hyperphysics.phy-astr.gsu.edu/hbase/math/stirling.html
109 negLogL += (mc - data + data * std::log(data/mc));
110 }
111 else if(mc > 0 && data == 0) negLogL += mc;
112
113 return negLogL;
114}

116// *************************
117// data is data, mc is mc, w2 is Sum(w_{i}^2) (sum of weights squared), which is sigma^2_{MC stats}
118double samplePDFBase::getTestStatLLH(const double data, const double mc, const double w2) const {
119// *************************
120
121 // Need some MC
122 if (mc == 0) return 0.0;
123
124 // The MC used in the likelihood calculation
125 // Is allowed to be changed by Barlow Beeston beta parameters
126 double newmc = mc;
127
128 switch (fTestStatistic)
129 {
130 //CW: Not full Barlow-Beeston or what is referred to as "light": we're not introducing any more parameters
131 // Assume the MC has a Gaussian distribution around generated
132 // As in https://arxiv.org/abs/1103.0354 eq 10, 11
133 //CW: Calculate the Barlow-Beeston likelihood contribution from MC statistics
134 // Assumes the beta scaling parameters are Gaussian distributed
135 // Follows arXiv:1103.0354 section 5 and equation 8, 9, 10, 11 on page 4/5
136 // Essentially solves equation 11
137 case (kBarlowBeeston):
138 {
139 // The penalty from MC statistics using Conways approach (https://cds.cern.ch/record/1333496?)
140 double penalty = 0;
141 // Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
142 const double fractional = std::sqrt(w2)/mc;
143 // b in quadratic equation
144 const double temp = mc*fractional*fractional-1;
145 // b^2 - 4ac in quadratic equation
146 const double temp2 = temp*temp + 4*data*fractional*fractional;
147 if (temp2 < 0) {
148 MACH3LOG_ERROR("Negative square root in Barlow Beeston coefficient calculation!");
149 throw MaCh3Exception(__FILE__ , __LINE__ );
150 }
151 // Solve for the positive beta
152 const double beta = (-1*temp+sqrt(temp2))/2.;
153 newmc = mc*beta;
154 // And penalise the movement in beta relative the mc uncertainty
155 if (fractional > 0) penalty = (beta-1)*(beta-1)/(2*fractional*fractional);
156 else penalty = 0;
157
158 // Calculate the new Poisson likelihood
159 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
160 // For the Poisson likelihood, this is just the usual calculation
161 // For IceCube likelihood, we calculate it later
162 double stat = 0;
163 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
164 if (data == 0) stat = newmc;
165 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
166
167 // Return the statistical contribution and penalty
168 return stat+penalty;
169 }
170 break;
171 //KS: Alternative calculation of Barlow-Beeston following Hans Dembinski and Ahmed Abdelmottele arXiv:2206.12346v2
172 case (kDembinskiAbdelmottele):
173 {
174 //KS: code follows authors implementation from:
175 //https://github.com/scikit-hep/iminuit/blob/059d06b00cae097ebf340b218b4eb57357111df8/src/iminuit/cost.py#L274-L300
176
177 //the so-called effective count
178 const double k = mc*mc / w2;
179 //Calculate beta which is scaling factor between true and generated MC
180 const double beta = (data + k) / (mc + k);
181
182 newmc = mc*beta;
183 // And penalise the movement in beta relative the mc uncertainty
184 const double penalty = k*beta-k+k*std::log(k/(k*beta));
185
186 // Calculate the new Poisson likelihood
187 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
188 // For the Poisson likelihood, this is just the usual calculation
189 // For IceCube likelihood, we calculate it later
190 double stat = 0;
191 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
192 if (data == 0) stat = newmc;
193 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
194
195 // Return the statistical contribution and penalty
196 return stat+penalty;
197 }
198 break;
199 //CW: Also try the IceCube likelihood
200 // It does not modify the MC content
201 // https://arxiv.org/abs/1901.04645
202 // Arg√ºelles, C.A., Schneider, A. & Yuan, T. J. High Energ. Phys. (2019) 2019: 30. https://doi.org/10.1007/JHEP06(2019)030
203 // We essentially construct eq 3.16 and take the logarithm
204 // in eq 3.16, mu is MC, sigma2 is w2, k is data
205 case (kIceCube):
206 {
207 double stat = 0.0;
208
209 // If there for some reason is 0 mc uncertainty, return the Poisson LLH
210 if (w2 == 0)
211 {
212 // Calculate the new Poisson likelihood
213 if (data == 0) stat = newmc;
214 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
215
216 return stat;
217 }
218 // Auxiliary variables
219 const long double b = mc/w2;
220 const long double a = mc*b+1;
221 const long double k = data;
222 // Use C99's implementation of log of gamma function to not be C++11 dependent
223 stat = double(-1*(a * logl(b) + lgammal(k+a) - lgammal(k+1) - ((k+a)*log1pl(b)) - lgammal(a)));
224
225 // Return the statistical contribution and penalty
226 return stat;
227 }
228 break;
229 //KS: Pearson works on assumption that event distribution in each bin is described by a Gaussian which in our case is not fulfilled for all bins, hence use it at your own risk
230 case (kPearson):
231 {
232 //KS: 2 is because this function returns -LLH not -2LLH
233 const double stat = (data-mc)*(data-mc)/(2*mc);
234
235 // Return the statistical
236 return stat;
237 }
238 break;
239 case (kPoisson):
240 {
241 //Just call getTestStatLLH which doesn't take in weights
242 //and is a Poisson likelihood comparison.
243 return getTestStatLLH(data, mc);//stat;
244 }
245 break;
246
247 default:
248 std::cerr << "Couldn't find TestStatistic " << fTestStatistic << " exiting!" << std::endl;
249 throw MaCh3Exception(__FILE__ , __LINE__ );
250 } // end switch
251}
