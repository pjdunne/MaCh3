200// Poisson likelihood calc for data and MC event rates
201double samplePDFBase::getTestStatLLH(const double data, const double mc) {
202// ***************************************************************************
203 // Need some MC
204 if(mc == 0) return 0.;
205
206 double negLogL = 0;
207 if(mc > 0 && data > 0)
208 {
209 //http://hyperphysics.phy-astr.gsu.edu/hbase/math/stirling.html
210 negLogL += (mc - data + data * std::log(data/mc));
211 }
212 else if(mc > 0 && data == 0) negLogL += mc;
213
214 return negLogL;
215}

217// *************************
218// data is data, mc is mc, w2 is Sum(w_{i}^2) (sum of weights squared), which is sigma^2_{MC stats}
219double samplePDFBase::getTestStatLLH(const double data, const double mc, const double w2) {
220// *************************
221
222 // Need some MC
223 if (mc == 0) return 0.0;
224
225 // The MC used in the likeliihood calculation
226 // Is allowed to be changed by Barlow Beeston beta parameters
227 double newmc = mc;
228
229 switch (fTestStatistic)
230 {
231 //CW: Not full Barlow-Beeston or what is referred to as "light": we're not introducing any more parameters
232 // Assume the MC has a Gaussian distribution around generated
233 // As in https://arxiv.org/abs/1103.0354 eq 10, 11
234 //CW: Calculate the Barlow-Beeston likelhood contribution from MC statistics
235 // Assumes the beta scaling parameters are Gaussian distributed
236 // Follows arXiv:1103.0354 section 5 and equation 8, 9, 10, 11 on page 4/5
237 // Essentially solves equation 11
238 case (kBarlowBeeston):
239 {
240 // The penalty from MC statistics using Conways approach (https://cds.cern.ch/record/1333496?)
241 double penalty = 0;
242 // Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
243 const double fractional = std::sqrt(w2)/mc;
244 // b in quadratic equation
245 const double temp = mc*fractional*fractional-1;
246 // b^2 - 4ac in quadratic equation
247 const double temp2 = temp*temp + 4*data*fractional*fractional;
248 if (temp2 < 0) {
249 std::cerr << "Negative square root in Barlow Beeston coefficient calculation!" << std::endl;
250 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
251 throw;
252 }
253 // Solve for the positive beta
254 const double beta = (-1*temp+sqrt(temp2))/2.;
255 newmc = mc*beta;
256 // And penalise the movement in beta relative the mc uncertainty
257 if (fractional > 0) penalty = (beta-1)*(beta-1)/(2*fractional*fractional);
258 else penalty = 0;
259
260 // Calculate the new Poisson likelihood
261 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
262 // For the Poisson likelihood, this is just the usual calculation
263 // For IceCube likelihood, we calculate it later
264 double stat = 0;
265 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
266 if (data == 0) stat = newmc;
267 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
268
269 // Return the statistical contribution and penalty
270 return stat+penalty;
271 }
272 break;
273 //KS: Alternative calculation of Barlow-Beeston following Hans Dembinski and Ahmed Abdelmottele arXiv:2206.12346v2
274 case (kDembinskiAbdelmottele):
275 {
276 //KS: code follows authors implementation from:
277 //https://github.com/scikit-hep/iminuit/blob/059d06b00cae097ebf340b218b4eb57357111df8/src/iminuit/cost.py#L274-L300
278
279 //the so-called effective count
280 const double k = mc*mc / w2;
281 //Calculate beta which is scaling factor between true and generated MC
282 const double beta = (data + k) / (mc + k);
283
284 newmc = mc*beta;
285 // And penalise the movement in beta relative the mc uncertainty
286 const double penalty = k*beta-k+k*std::log(k/(k*beta));
287
288 // Calculate the new Poisson likelihood
289 // For Barlow-Beeston newmc is modified, so can only calculate Poisson likelihood after Barlow-Beeston
290 // For the Poisson likelihood, this is just the usual calculation
291 // For IceCube likelihood, we calculate it later
292 double stat = 0;
293 // All likelihood calculations may use the bare Poisson likelihood, so calculate here
294 if (data == 0) stat = newmc;
295 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
296
297 // Return the statistical contribution and penalty
298 return stat+penalty;
299 }
300 break;
301 //CW: Also try the IceCube likelihood
302 // It does not modify the MC content
303 // https://arxiv.org/abs/1901.04645
304 // Argelles, C.A., Schneider, A. & Yuan, T. J. High Energ. Phys. (2019) 2019: 30. https://doi.org/10.1007/JHEP06(2019)030
305 // We essentially construct eq 3.16 and take the logarithm
306 // in eq 3.16, mu is MC, sigma2 is w2, k is data
307 case (kIceCube):
308 {
309 double stat = 0.0;
310
311 // If there for some reason is 0 mc uncertainty, return the Poisson LLH
312 if (w2 == 0)
313 {
314 // Calculate the new Poisson likelihood
315 if (data == 0) stat = newmc;
316 else if (newmc > 0) stat = newmc-data+data*std::log(data/newmc);
317
318 return stat;
319 }
320 // Auxillary variables
321 const long double b = mc/w2;
322 const long double a = mc*b+1;
323 const long double k = data;
324 // Use C99's implementation of log of gamma function to not be C++11 dependent
325 stat = -1*(a * logl(b) + lgammal(k+a) - lgammal(k+(long double)1) - ((k+a)*log1pl(b)) - lgammal(a));
326
327 // Return the statistical contribution and penalty
328 return stat;
329 }
330 break;
331 //KS: Pearson works on assumption that event distribution in each bin is described by a Gaussian which in our case is not fulfilled for all bins, hence use it at your own risk
332 case (kPearson):
333 {
334 //KS: 2 is because this function returns -LLH not -2LLH
335 const double stat = (data-mc)*(data-mc)/(2*mc);
336
337 // Return the statistical
338 return stat;
339 }
340 break;
341 case (kPoisson):
342 {
343 //Just call getTestStatLLH which doesn't take in weights
344 //and is a poisson likelihood comparison.
345 return getTestStatLLH(data, mc);//stat;
346 }
347 break;
348
349 default:
350 std::cerr << "Couldn't find TestStatistic " << fTestStatistic << " exiting!" << std::endl;
351 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
352 throw;
353 } // end switch
354}
