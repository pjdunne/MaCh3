153inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
154// **************************************************
155 RespFuncType Type = kRespFuncTypes;
156 switch(i) {
157 // TSpline3 (third order spline in ROOT)
158 case kTSpline3:
159 Type = kTSpline3_red;
160 break;
161 case kLinear:
162 Type = kTSpline3_red;
163 break;
164 case kMonotonic:
165 Type = kTSpline3_red;
166 break;
167 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)
168 case kAkima:
169 Type = kTSpline3_red;
170 break;
171 case kLinearFunc:
172 Type = kTF1_red;
173 break;
174 default:
175 std::cerr << "UNKNOWN SPLINE INTERPOLATION SPECIFIED!" << std::endl;
176 std::cerr << "You gave " << i << std::endl;
177 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
178 throw;
179 }
180 return Type;
181}
