125inline std::string SplineInterpolation_ToString(const SplineInterpolation i) {
126// **************************************************
127 std::string name = "";
128 switch(i) {
129 // TSpline3 (third order spline in ROOT)
130 case kTSpline3:
131 name = "TSpline3";
132 break;
133 case kLinear:
134 name = "Linear";
135 break;
136 case kMonotonic:
137 name = "Monotonic";
138 break;
139 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)

141 name = "Akima";
142 break;

144 std::cerr << "UNKNOWN SPLINE INTERPOLATION SPECIFIED!" << std::endl;
145 std::cerr << "You gave " << i << std::endl;
146 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
147 throw;
148 }
149 return name;
