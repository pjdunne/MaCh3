121inline std::string SplineInterpolation_ToString(const SplineInterpolation i) {
122// **************************************************
123 std::string name = "";
124 switch(i) {
125 // TSpline3 (third order spline in ROOT)
126 case kTSpline3:
127 name = "TSpline3";
128 break;
129 case kLinear:
130 name = "Linear";
131 break;
132 case kMonotonic:
133 name = "Monotonic";
134 break;
135 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)
136 case kAkima:
137 name = "Akima";
138 break;
139 default:
140 std::cerr << "UNKNOWN SPLINE INTERPOLATION SPECIFIED!" << std::endl;
141 std::cerr << "You gave " << i << std::endl;
142 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
143 throw;
144 }
145 return name;
146}
