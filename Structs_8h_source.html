138inline std::string GetTF1(const SplineInterpolation i) {
139 // **************************************************
140 std::string Func = "";
141 switch(i) {
142 case SplineInterpolation::kLinearFunc:
143 Func = "([1]+[0]*x)";
144 break;
145 default:
146 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
147 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
148 throw MaCh3Exception(__FILE__ , __LINE__ );
149 }
150 return Func;
151}

153// **************************************************
156inline RespFuncType SplineInterpolation_ToRespFuncType(const SplineInterpolation i) {
157// **************************************************
158 RespFuncType Type = kRespFuncTypes;
159 switch(i) {
160 // TSpline3 (third order spline in ROOT)
161 case SplineInterpolation::kTSpline3:
162 Type = RespFuncType::kTSpline3_red;
163 break;
164 case SplineInterpolation::kLinear:
165 Type = RespFuncType::kTSpline3_red;
166 break;
167 case SplineInterpolation::kMonotonic:
168 Type = RespFuncType::kTSpline3_red;
169 break;
170 // (Experimental) Akima_Spline (crd order spline which is allowed to be discontinuous in 2nd deriv)
171 case SplineInterpolation::kAkima:
172 Type = RespFuncType::kTSpline3_red;
173 break;
174 case SplineInterpolation::kLinearFunc:
175 Type = RespFuncType::kTF1_red;
176 break;
177 default:
178 MACH3LOG_ERROR("UNKNOWN SPLINE INTERPOLATION SPECIFIED!");
179 MACH3LOG_ERROR("You gave {}", static_cast<int>(i));
180 throw MaCh3Exception(__FILE__ , __LINE__ );
181 }
182 return Type;
183}
