54 //Set param type

64 //to the Systematic index i.e. the counter for things like _fDetID and _fDetID

66 ParamCounter[SystType::kSpline]++;

523// KS: Check if matrix is correctly initialised
524void covarianceXsec::CheckCorrectInitialisation() {
525// ********************************************
526 // KS: Lambda Function which simply checks if there are no duplicates in std::vector
527 auto CheckForDuplicates = [](const std::vector<std::string>& names, const std::string& nameType) {
528 std::unordered_map<std::string, size_t> seenStrings;
529 for (size_t i = 0; i < names.size(); ++i) {
530 const auto& name = names[i];
531 if (seenStrings.find(name) != seenStrings.end()) {
532 size_t firstIndex = seenStrings[name];
