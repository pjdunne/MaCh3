579 //DB Catch negative func weights and skip any event with a negative event. Previously we would set weight to zero and continue but that is inefficient
580 if (funcweight <= 0.){
581 MCSamples[iSample].xsec_w[iEvent] = 0.;
582 continue;
583 }
584
585 MCSamples[iSample].xsec_w[iEvent] = splineweight*normweight*funcweight;
586
587 totalweight = GetEventWeight(iSample, iEvent);
588
589 //DB Catch negative weights and skip any event with a negative event
590 if (totalweight <= 0.){
591 MCSamples[iSample].xsec_w[iEvent] = 0.;
592 continue;
593 }
594
595 //DB Switch on BinningOpt to allow different binning options to be implemented
596 //The alternative would be to have inheritance based on BinningOpt
597 double XVar = (*(MCSamples[iSample].x_var[iEvent]));
598
599 //DB Commented out by default but if we ever want to consider shifts in theta this will be needed
600 //double YVar = MCSamples[iSample].rw_theta[iEvent];
601 //ETA - this would actually be with (*(MCSamples[iSample].y_var[iEvent])) and done extremely
602 //similarly to XVar now
603
604 //DB Find the relevant bin in the PDF for each event
605 int XBinToFill = -1;
606 int YBinToFill = MCSamples[iSample].NomYBin[iEvent];
607
608 //DB Check to see if momentum shift has moved bins
609 //DB - First, check to see if the event is still in the nominal bin
610 if (XVar < MCSamples[iSample].rw_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_xbinedge[iEvent]) {
611 XBinToFill = MCSamples[iSample].NomXBin[iEvent];
612 }
613 //DB - Second, check to see if the event is outside of the binning range and skip event if it is
614 else if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
615 MACH3LOG_WARN("XVAR BEYOND BIN EDGES!!");
616 continue;
617 }
618 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
619 //Shifted down one bin from the event bin at nominal
620 else if (XVar < MCSamples[iSample].rw_lower_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_lower_xbinedge[iEvent]) {
621 XBinToFill = MCSamples[iSample].NomXBin[iEvent]-1;
622 }
623 //Shifted up one bin from the event bin at nominal
624 else if (XVar < MCSamples[iSample].rw_upper_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_upper_xbinedge[iEvent]) {
625 XBinToFill = MCSamples[iSample].NomXBin[iEvent]+1;
626 }
627 //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
628 else {
629 for(unsigned int iBin=0;iBin<(XBinEdges.size()-1);iBin++) {
630 if (XVar >= XBinEdges[iBin] && XVar < XBinEdges[iBin+1]) {
631 XBinToFill = iBin;
632 }
633 }
634 }
635
636 //ETA - we can probably remove this final if check on the -1?
637 //Maybe we can add an overflow bin to the array and assign any events to this bin?
638 //Might save us an extra if call?
639 //DB Fill relevant part of thread array
640 if (XBinToFill != -1 && YBinToFill != -1) {
641 samplePDFFD_array_private[YBinToFill][XBinToFill] += totalweight;
642 samplePDFFD_array_private_w2[YBinToFill][XBinToFill] += totalweight*totalweight;
643 }
644 }
645 }
646 //End of Calc Weights and fill Array
647 //==================================================
648 // DB Copy contents of 'samplePDFFD_array_private' into 'samplePDFFD_array' which can then be used in GetLikelihood
649 for (size_t yBin = 0; yBin < nYBins; ++yBin) {
650 for (size_t xBin = 0; xBin < nXBins; ++xBin) {
651 #pragma omp atomic
652 samplePDFFD_array[yBin][xBin] += samplePDFFD_array_private[yBin][xBin];
653 #pragma omp atomic
654 samplePDFFD_array_w2[yBin][xBin] += samplePDFFD_array_private_w2[yBin][xBin];
655 }
656 }
657
658 for (size_t yBin = 0; yBin < nYBins; ++yBin) {
659 delete[] samplePDFFD_array_private[yBin];
660 delete[] samplePDFFD_array_private_w2[yBin];
661 }
662 delete[] samplePDFFD_array_private;
663 delete[] samplePDFFD_array_private_w2;
664 } //end of parallel region

750void samplePDFFDBase::CalcXsecNormsBins(int iSample){
751 FarDetectorCoreInfo *fdobj = &MCSamples[iSample];
752 #ifdef DEBUG
753 std::vector<int> VerboseCounter(xsec_norms.size(), 0);
754 #endif
755 for(int iEvent=0; iEvent < fdobj->nEvents; ++iEvent){
756 std::vector< int > XsecBins = {};
757 if (XsecCov) {
758 // Skip oscillated NC events
759 // Not strictly needed, but these events don't get included in oscillated predictions, so
760 // no need to waste our time calculating and storing information about xsec parameters
761 // that will never be used.
762 if (fdobj->isNC[iEvent] && fdobj->signal) {
763 MACH3LOG_TRACE("Event {}, missed NC/signal check", iEvent);
764 continue;
765 } //DB Abstract check on MaCh3Modes to determine which apply to neutral current
766 for (std::vector<XsecNorms4>::iterator it = xsec_norms.begin(); it != xsec_norms.end(); ++it) {
767 //Now check that the target of an interaction matches with the normalisation parameters
768 bool TargetMatch = false;
769 //If no target specified then apply to all modes
770 if ((*it).targets.size()==0) {
771 TargetMatch=true;
772 } else {
773 for (unsigned iTarget=0;iTarget<(*it).targets.size();iTarget++) {
774 if ((*it).targets.at(iTarget)== *(fdobj->Target[iEvent])) {
775 TargetMatch=true;
776 }
777 }
778 }
779 if (!TargetMatch) {
780 MACH3LOG_TRACE("Event {}, missed target check ({}) for dial {}", iEvent, *(fdobj->Target[iEvent]), (*it).name);
781 continue;
782 }
783
784 //Now check that the neutrino flavour in an interaction matches with the normalisation parameters
785 bool FlavourMatch=false;
786 //If no mode specified then apply to all modes
787 if ((*it).pdgs.size()==0) {
788 FlavourMatch=true;
789 } else {
790 for (unsigned iPDG=0;iPDG<(*it).pdgs.size();iPDG++) {
791 if ((*it).pdgs.at(iPDG)== (*fdobj->nupdg[iEvent])) {
792 FlavourMatch=true;
793 }
794 }
795 }
796 if (!FlavourMatch) {
797 MACH3LOG_TRACE("Event {}, missed PDG check ({}) for dial {}", iEvent,(*fdobj->nupdg[iEvent]), (*it).name);
798 continue;
799 }
800
801 //Now check that the unoscillated neutrino flavour in an interaction matches with the normalisation parameters
802 bool FlavourUnoscMatch=false;
803 //If no mode specified then apply to all modes
804 if ((*it).preoscpdgs.size()==0) {
805 FlavourUnoscMatch=true;
806 } else {
807 for (unsigned iPDG=0;iPDG<(*it).preoscpdgs.size();iPDG++) {
808 if ((*it).preoscpdgs.at(iPDG) == (*fdobj->nupdgUnosc[iEvent])) {
809 FlavourUnoscMatch=true;
810 }
811 }
812 }
813 if (!FlavourUnoscMatch){
814 MACH3LOG_TRACE("Event {}, missed FlavourUnosc check ({}) for dial {}", iEvent,(*fdobj->nupdgUnosc[iEvent]), (*it).name);
815 continue;
816 }
817
818 //Now check that the mode of an interaction matches with the normalisation parameters
819 bool ModeMatch=false;
820 //If no mode specified then apply to all modes
821 if ((*it).modes.size()==0) {
822 ModeMatch=true;
823 } else {
824 for (unsigned imode=0;imode<(*it).modes.size();imode++) {
825 if ((*it).modes.at(imode)== *(fdobj->mode[iEvent])) {
826 ModeMatch=true;
827 }
828 }
829 }
830 if (!ModeMatch) {
831 MACH3LOG_TRACE("Event {}, missed Mode check ({}) for dial {}", iEvent, *(fdobj->mode[iEvent]), (*it).name);
832 continue;
833 }
834
835 //Now check whether the norm has kinematic bounds
836 //i.e. does it only apply to events in a particular kinematic region?
837 bool IsSelected = true;
838 if ((*it).hasKinBounds) {
839 for (unsigned int iKinematicParameter = 0 ; iKinematicParameter < (*it).KinematicVarStr.size() ; ++iKinematicParameter ) {
840 if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) <= (*it).Selection[iKinematicParameter][0]) {
841 IsSelected = false;
842 MACH3LOG_TRACE("Event {}, missed Kinematic var check ({}) for dial {}", iEvent, (*it).KinematicVarStr[iKinematicParameter], (*it).name);
843 continue;
844 }
845 else if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) > (*it).Selection[iKinematicParameter][1]) {
846 MACH3LOG_TRACE("Event {}, missed Kinematic var check ({}) for dial {}", iEvent, (*it).KinematicVarStr[iKinematicParameter], (*it).name);
847 IsSelected = false;
848 continue;
849 }
850 }
851 }
852 //Need to then break the event loop
853 if(!IsSelected){
854 MACH3LOG_TRACE("Event {}, missed Kinematic var check for dial {}", iEvent, (*it).name);
855 continue;
856 }
857 // Now set 'index bin' for each normalisation parameter
858 // All normalisations are just 1 bin for 2015, so bin = index (where index is just the bin for that normalisation)
859 int bin = (*it).index;
860
861 //If syst on applies to a particular detector
862 if ((XsecCov->GetParDetID(bin) & SampleDetID)==SampleDetID) {
863 XsecBins.push_back(bin);
864 MACH3LOG_TRACE("Event {}, will be affected by dial {}", iEvent, (*it).name);
865 #ifdef DEBUG
866 VerboseCounter[std::distance(xsec_norms.begin(), it)]++;
867 #endif
868 }
869 } // end iteration over xsec_norms
870 } // end if (xsecCov)
871 fdobj->xsec_norms_bins[iEvent]=XsecBins;
872 }//end loop over events
873 #ifdef DEBUG
874 MACH3LOG_DEBUG("Channel {}", iSample);
875 MACH3LOG_DEBUG("");
876 for (std::size_t i = 0; i < xsec_norms.size(); ++i) {
877 const auto& norm = xsec_norms[i];
878 double eventRatio = static_cast<double>(VerboseCounter[i]) / static_cast<double>(fdobj->nEvents);
879
880 MACH3LOG_DEBUG(" Param {:<15}, affects {:<8} events ({:>6.2f}%) ",
881 XsecCov->GetParFancyName(norm.index), VerboseCounter[i], eventRatio);
882 }
883 MACH3LOG_DEBUG("");
884 #endif
885}
