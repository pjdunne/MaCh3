593 //DB SKDet Syst
594 //As weights were skdet::fParProp, and we use the non-shifted erec, we might as well cache the corresponding fParProp index for each event and the pointer to it
595
596 if(SplineHandler){
597 splineweight *= CalcXsecWeightSpline(iSample, iEvent);
598 }
599 //DB Catch negative spline weights and skip any event with a negative event. Previously we would set weight to zero and continue but that is inefficient
600 if (splineweight <= 0.){
601 MCSamples[iSample].xsec_w[iEvent] = 0.;
602 continue;
603 }
604
605 normweight *= CalcXsecWeightNorm(iSample, iEvent);
606 //DB Catch negative norm weights and skip any event with a negative event. Previously we would set weight to zero and continue but that is inefficient
607 if (normweight <= 0.){
608 MCSamples[iSample].xsec_w[iEvent] = 0.;
609 continue;
610 }
611
612 funcweight = CalcXsecWeightFunc(iSample,iEvent);
613 //DB Catch negative func weights and skip any event with a negative event. Previously we would set weight to zero and continue but that is inefficient
614 if (funcweight <= 0.){
615 MCSamples[iSample].xsec_w[iEvent] = 0.;
616 continue;
617 }
618
619 MCSamples[iSample].xsec_w[iEvent] = splineweight*normweight*funcweight;
620
621 totalweight = GetEventWeight(iSample, iEvent);
622
623 //DB Catch negative weights and skip any event with a negative event
624 if (totalweight <= 0.){
625 //std::cout << "NEGATIVE WEIGHT!!" << std::endl;
626 MCSamples[iSample].xsec_w[iEvent] = 0.;
627 continue;
628 }
629
630 //DB Switch on BinningOpt to allow different binning options to be implemented
631 //The alternative would be to have inheritance based on BinningOpt
632 double XVar = (*(MCSamples[iSample].x_var[iEvent]));
633
634 //DB Commented out by default but if we ever want to consider shifts in theta this will be needed
635 //double YVar = MCSamples[iSample].rw_theta[iEvent];
636 //ETA - this would actually be with (*(MCSamples[iSample].y_var[iEvent])) and done extremely
637 //similarly to XVar now
638
639 //DB Find the relevant bin in the PDF for each event
640 int XBinToFill = -1;
641 int YBinToFill = MCSamples[iSample].NomYBin[iEvent];
642
643 //DB Check to see if momentum shift has moved bins
644 //DB - First, check to see if the event is still in the nominal bin
645 if (XVar < MCSamples[iSample].rw_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_xbinedge[iEvent]) {
646 XBinToFill = MCSamples[iSample].NomXBin[iEvent];
647 }
648 //DB - Second, check to see if the event is outside of the binning range and skip event if it is
649 else if (XVar < XBinEdges[0] || XVar >= XBinEdges[nXBins]) {
650 std::cout << "XVAR BEYOND BIN EDGES!!" << std::endl;
651 continue;
652 }
653 //DB - Thirdly, check the adjacent bins first as Eb+CC+EScale shifts aren't likely to move an Erec more than 1bin width
654 //Shifted down one bin from the event bin at nominal
655 else if (XVar < MCSamples[iSample].rw_lower_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_lower_lower_xbinedge[iEvent]) {
656 XBinToFill = MCSamples[iSample].NomXBin[iEvent]-1;
657 }
658 //Shifted up one bin from the event bin at nominal
659 else if (XVar < MCSamples[iSample].rw_upper_upper_xbinedge[iEvent] && XVar >= MCSamples[iSample].rw_upper_xbinedge[iEvent]) {
660 XBinToFill = MCSamples[iSample].NomXBin[iEvent]+1;
661 }
662 //DB - If we end up in this loop, the event has been shifted outside of its nominal bin, but is still within the allowed binning range
663 else {
664 for(unsigned int iBin=0;iBin<(XBinEdges.size()-1);iBin++) {
665 if (XVar >= XBinEdges[iBin] && XVar < XBinEdges[iBin+1]) {
666 XBinToFill = iBin;
667 }
668 }
669 }
670
671 //ETA - we can probably remove this final if check on the -1?
672 //Maybe we can add an overflow bin to the array and assign any events to this bin?
673 //Might save us an extra if call?
674 //DB Fill relevant part of thread array
675 if (XBinToFill != -1 && YBinToFill != -1) {
676 samplePDFFD_array_private[YBinToFill][XBinToFill] += totalweight;
677 samplePDFFD_array_private_w2[YBinToFill][XBinToFill] += totalweight*totalweight;
678 }
679 }
680 }
681
682 //End of Calc Weights and fill Array
683 //==================================================
684 // DB Copy contents of 'samplePDFFD_array_private' into 'samplePDFFD_array' which can then be used in GetLikelihood
685 for (size_t yBin=0;yBin<nYBins;yBin++) {
686 for (size_t xBin=0;xBin<nXBins;xBin++) {
687#pragma omp atomic
688 samplePDFFD_array[yBin][xBin] += samplePDFFD_array_private[yBin][xBin];
689#pragma omp atomic
690 samplePDFFD_array_w2[yBin][xBin] += samplePDFFD_array_private_w2[yBin][xBin];
691 }
692 }
693
694 for (size_t yBin=0;yBin<nYBins;yBin++) {
695 delete[] samplePDFFD_array_private[yBin];
696 delete[] samplePDFFD_array_private_w2[yBin];
697 }
698 delete[] samplePDFFD_array_private;
699 delete[] samplePDFFD_array_private_w2;
700 } //end of parallel region
701}

792//A way to check whether a normalisation parameter applies to an event or not
793void samplePDFFDBase::CalcXsecNormsBins(int iSample){
794
795 FarDetectorCoreInfo *fdobj = &MCSamples[iSample];
796 #ifdef DEBUG
797 std::vector<int> VerboseCounter(xsec_norms.size(), 0);
798 #endif
799 for(int iEvent=0; iEvent < fdobj->nEvents; ++iEvent){
800 std::vector< int > XsecBins = {};
801 if (XsecCov) {
802 for (std::vector<XsecNorms4>::iterator it = xsec_norms.begin(); it != xsec_norms.end(); ++it) {
803 // Skip oscillated NC events
804 // Not strictly needed, but these events don't get included in oscillated predictions, so
805 // no need to waste our time calculating and storing information about xsec parameters
806 // that will never be used.
807 if (fdobj->isNC[iEvent] && fdobj->signal) {continue;} //DB Abstract check on MaCh3Modes to determine which apply to neutral current
808
809 //Now check that the target of an interaction matches with the normalisation parameters
810 bool TargetMatch=false;
811 //If no target specified then apply to all modes
812 if ((*it).targets.size()==0) {
813 TargetMatch=true;
814 } else {
815 for (unsigned iTarget=0;iTarget<(*it).targets.size();iTarget++) {
816 if ((*it).targets.at(iTarget)== *(fdobj->Target[iEvent])) {
817 TargetMatch=true;
818 }
819 }
820 }
821 if (!TargetMatch) {continue;}
822
823 //Now check that the neutrino flavour in an interaction matches with the normalisation parameters
824 bool FlavourMatch=false;
825 //If no mode specified then apply to all modes
826 if ((*it).pdgs.size()==0) {
827 FlavourMatch=true;
828 } else {
829 for (unsigned iPDG=0;iPDG<(*it).pdgs.size();iPDG++) {
830 if ((*it).pdgs.at(iPDG)== (*fdobj->nupdg[iEvent])) {
831 FlavourMatch=true;
832 }
833 }
834 }
835 if (!FlavourMatch){continue;}
836
837 //Now check that the unoscillated neutrino flavour in an interaction matches with the normalisation parameters
838 bool FlavourUnoscMatch=false;
839 //If no mode specified then apply to all modes
840 if ((*it).preoscpdgs.size()==0) {
841 FlavourUnoscMatch=true;
842 } else {
843 for (unsigned iPDG=0;iPDG<(*it).preoscpdgs.size();iPDG++) {
844 if ((*it).preoscpdgs.at(iPDG) == (*fdobj->nupdgUnosc[iEvent])) {
845 FlavourUnoscMatch=true;
846 }
847 }
848 }
849 if (!FlavourUnoscMatch){continue;}
850
851 //Now check that the mode of an interaction matches with the normalisation parameters
852 bool ModeMatch=false;
853 //If no mode specified then apply to all modes
854 if ((*it).modes.size()==0) {
855 ModeMatch=true;
856 } else {
857 for (unsigned imode=0;imode<(*it).modes.size();imode++) {
858 if ((*it).modes.at(imode)== *(fdobj->mode[iEvent])) {
859 ModeMatch=true;
860 }
861 }
862 }
863 if (!ModeMatch) {continue;}
864
865 //Now check whether the norm has kinematic bounds
866 //i.e. does it only apply to events in a particular kinematic region?
867 bool IsSelected = true;
868 if ((*it).hasKinBounds) {
869 for (unsigned int iKinematicParameter = 0 ; iKinematicParameter < (*it).KinematicVarStr.size() ; ++iKinematicParameter ) {
870 if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) <= (*it).Selection[iKinematicParameter][0]) {
871 IsSelected = false;
872 continue;
873 }
874 else if (ReturnKinematicParameter((*it).KinematicVarStr[iKinematicParameter], iSample, iEvent) > (*it).Selection[iKinematicParameter][1]) {
875 IsSelected = false;
876 continue;
877 }
878 }
879 }
880 //Need to then break the event loop
881 if(!IsSelected){
882 continue;
883 }
884 // Now set 'index bin' for each normalisation parameter
885 // All normalisations are just 1 bin for 2015, so bin = index (where index is just the bin for that normalisation)
886 int bin = (*it).index;
887
888 //If syst on applies to a particular detector
889 if ((XsecCov->GetParDetID(bin) & SampleDetID)==SampleDetID) {
890 XsecBins.push_back(bin);
891 #ifdef DEBUG
892 VerboseCounter[std::distance(xsec_norms.begin(), it)]++;
893 #endif
894 }
895 } // end iteration over xsec_norms
896 } // end if (xsecCov)
897 fdobj->xsec_norms_bins[iEvent]=XsecBins;
898 }//end loop over events
899 #ifdef DEBUG
900 MACH3LOG_INFO("Channel {}", iSample);
901 MACH3LOG_INFO("");
902 for (std::size_t i = 0; i < xsec_norms.size(); ++i) {
903 const auto& norm = xsec_norms[i];
904 double eventRatio = static_cast<double>(VerboseCounter[i]) / static_cast<double>(fdobj->nEvents);
905
906 MACH3LOG_INFO(" Param {:<15}, affects {:<8} events ({:>6.2f}%) ",
907 XsecCov->GetParFancyName(norm.index), VerboseCounter[i], eventRatio);
908 }
909 MACH3LOG_INFO("");
910 #endif
911
912 return;
913}
