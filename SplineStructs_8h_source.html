51inline void ApplyKnotWeightCap(TGraph* xsecgraph, int splineParsIndex, covarianceXsec* XsecCov) {
52// ***************************************************************************
53 if(xsecgraph == NULL){
54 MACH3LOG_ERROR("ERROR: hmmm looks like you're trying to apply capping for spline parameter {} but it hasn't been set in xsecgraph yet", XsecCov->GetParFancyName(splineParsIndex));
55 throw MaCh3Exception(__FILE__ , __LINE__ );
56 }
57
58 // EM: cap the weights of the knots if specified in the config
59 if(
60 (XsecCov->GetParSplineKnotUpperBound(splineParsIndex) != 9999)
61 || (XsecCov->GetParSplineKnotLowerBound(splineParsIndex) != -9999))
62 {
63 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
64 double x,y;
65
66 // EM: get the x and y of the point. Also double check that the requested point was valid just to be super safe
67 if( xsecgraph->GetPoint(knotId, x, y) == -1) {
68 MACH3LOG_ERROR("Invalid knot requested: {}", knotId);
69 throw MaCh3Exception(__FILE__ , __LINE__ );
70 }
71
72 y = std::min(y, XsecCov->GetParSplineKnotUpperBound(splineParsIndex));
73 y = std::max(y, XsecCov->GetParSplineKnotLowerBound(splineParsIndex));
74
75 xsecgraph->SetPoint(knotId, x, y);
76 }
77
78 // EM: check if our cap made the spline flat, if so we set to 1 knot to avoid problems later on
79 bool isFlat = true;
80
81 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
82 double x,y;
83
84 // EM: get the x and y of the point. Also double check that the requested point was valid just to be super safe
85 if( xsecgraph->GetPoint(knotId, x, y) == -1) {
86 MACH3LOG_ERROR("Invalid knot requested: {}", knotId);
87 throw MaCh3Exception(__FILE__ , __LINE__ );
88 }
89
90 if(std::abs(y - 1.0) > 1e-5) isFlat = false;
91 }
92
93 if(isFlat){

312 Par[i][2] = M3::float_t(d);

316 // Not the most elegant way: use TSpline3 object but set coefficients to zero and recalculate spline points; the smart way (but more human intensive) would be to save memory here and simply not store the zeros at all
317 // Get which parameters should be linear from the fit manager
318 // Convert the spline number to global xsec parameter
319 // Loop over the splines points
320 // KS: kLinearFunc should be used with TF1, this is just as safety
321 else if(InterPolation == kLinear || InterPolation == kLinearFunc)
322 {

324 // 3 is the size of the TSpline3 coefficients
325 Par[k] = new M3::float_t[3];

329 double tempb = (y2-y1)/(x2-x1);

334 Par[k][1] = M3::float_t(0);
335 Par[k][2] = M3::float_t(0);
336 }

338 //EM: Akima spline is similar to regular cubic spline but is allowed to be discontinuous in 2nd derivative and coefficients in any segment
339 // only depend on th 2 nearest points on either side
340 else if(InterPolation == kAkima)
341 {
342 // get the knot values for the spline
343 for (int i = 0; i < nPoints; ++i) {
344 // 3 is the size of the TSpline3 coefficients
345 Par[i] = new M3::float_t[3];
346
347 double x = -999.99, y = -999.99;
348 spline->GetKnot(i, x, y);
349
350 XPos[i] = M3::float_t(x);
351 YResp[i] = M3::float_t(y);
352 }
353
354 M3::float_t* mvals = new M3::float_t[nPoints + 3];
355 M3::float_t* svals = new M3::float_t[nPoints + 1];
356
357 for (int i = -2; i <= nPoints; ++i) {
358 // if segment is first or last or 2nd to first or last, needs to be dealt with slightly differently;
359 // need to estimate the values for additinal points which would lie outside of the spline
360 if(i ==-2){
361 mvals[i+2] = M3::float_t(3.0 * (YResp[1] - YResp[0]) / (XPos[1] - XPos[0]) - 2.0*(YResp[2] - YResp[1]) / (XPos[2] - XPos[1]));
362 }
363 else if(i==-1){
364 mvals[i+2] = M3::float_t(2.0 * (YResp[1] - YResp[0]) / (XPos[1] - XPos[0]) - (YResp[2] - YResp[1]) / (XPos[2] - XPos[1]));
365 }
366 else if(i==nPoints){
367 mvals[i+2] = M3::float_t(3.0 * (YResp[nPoints-1] - YResp[nPoints-2]) / (XPos[nPoints-1] - XPos[nPoints-2]) - 2.0*(YResp[nPoints-2] - YResp[nPoints-3]) / (XPos[nPoints-2] - XPos[nPoints-3]));
368 }
369 else if(i == nPoints - 1){
370 mvals[i+2] = M3::float_t(2.0 * (YResp[nPoints-1] - YResp[nPoints-2]) / (XPos[nPoints-1] - XPos[nPoints-2]) - (YResp[nPoints-2] - YResp[nPoints-3]) / (XPos[nPoints-2] - XPos[nPoints-3]));
371 }
372 //standard internal segment
373 else{
374 mvals[i+2] = (YResp[i+1] - YResp[i])/ (XPos[i+1] - XPos[i]);
375 }
376 }
377
378 for(int i =2; i<=nPoints+2; i++){
379 if (std::abs(mvals[i+1] - mvals[i]) + std::abs(mvals[i-1] - mvals[i-2]) != 0.0){
380 svals[i-2] = (std::abs(mvals[i+1] - mvals[i]) * mvals[i-1] + std::abs(mvals[i-1] - mvals[i-2]) *mvals[i]) / (std::abs(mvals[i+1] - mvals[i]) + std::abs(mvals[i-1] - mvals[i-2]));
381 }
382 else{svals[i-2] = mvals[i];}
383 }
384
385 // calculate the coefficients for the spline
386 for(int i = 0; i <nPoints; i++){
387 M3::float_t b, c, d = M3::float_t(-999.999);
388
389 b = svals[i];
390 c = M3::float_t(3.0* (YResp[i+1] - YResp[i]) / (XPos[i+1] - XPos[i]) -2.0 *svals[i] - svals[i +1]) /(XPos[i+1] - XPos[i]);
391 d = M3::float_t((svals[i + 1] +svals[i]) - 2.0*(YResp[i+1] - YResp[i]) / (XPos[i+1] - XPos[i])) / ((XPos[i+1] - XPos[i]) * (XPos[i+1] - XPos[i]));
392
393 Par[i][0] = b;
394 Par[i][1] = c;
395 Par[i][2] = d;
396 }
397
398 // check the input spline for linear segments, if there are any then overwrite the calculated coefficients
399 // this will pretty much only ever be the case if they are set to be linear in samplePDFND i.e. the user wants it to be linear
400 for(int i = 0; i <nPoints-1; i++){
401 double x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
402 spline->GetCoeff(i, x, y, b, c, d);
403
404 if((c == 0.0 && d == 0.0)){
405 Par[i][0] = M3::float_t(b);
406 Par[i][1] = M3::float_t(0.0);
407 Par[i][2] = M3::float_t(0.0);
408 }
409 }

414 // must be between its two nearest knots, DOES NOT make the entire spline monotonic, only the segments
415 else if(InterPolation == kMonotonic)
416 {
417 // values of the secants at each point (for calculating monotone spline)
418 M3::float_t * Secants = new M3::float_t[nPoints -1];
419 // values of the tangens at each point (for calculating monotone spline)
420 M3::float_t * Tangents = new M3::float_t[nPoints];
421
422 // get the knot values for the spline
423 for (int i = 0; i < nPoints; ++i) {
424 // 3 is the size of the TSpline3 coefficients
425 Par[i] = new M3::float_t[3];
426
427 double x = -999.99, y = -999.99;
428 spline->GetKnot(i, x, y);
429
430 XPos[i] = M3::float_t(x);
431 YResp[i] = M3::float_t(y);
432
433 Tangents[i] = 0.0;
434 }
435
436 // deal with the case of two points (just do linear interpolation between them)

438 Par[0][0] = (YResp[1] - YResp[0]) / (XPos[1] - XPos[0]);

440 Par[0][2] = 0.0;

465 }
466 }
467

470 if (Secants[i] == 0.0){
471 Tangents[i] = 0.0;
472 Tangents[i+1] = 0.0;
473 }
474

476 alpha = Tangents[i] / Secants[i];
477 beta = Tangents[i+1] / Secants[i];
478
479 if (alpha <0.0){
480 Tangents[i] = 0.0;
481 }
482 if (beta < 0.0){
483 Tangents[i+1] = 0.0;
484 }
485

487 M3::float_t tau = M3::float_t(3.0 / std::sqrt(alpha * alpha + beta * beta));
488 Tangents[i] = tau * alpha * Secants[i];
489 Tangents[i+1] = tau * beta * Secants[i];
490 }
491 }
492 } // finished rescaling tangents
493 // fourth pass over knots to calculate the coefficients for the spline
494 M3::float_t dx;

496 M3::float_t b, c, d = M3::float_t(-999.999);
497 dx = XPos[i+1] - XPos[i];
498
499 b = Tangents[i] * dx;
500 c = M3::float_t(3.0* (YResp[i+1] - YResp[i]) -2.0 *dx * Tangents[i] - dx * Tangents[i +1]);
501 d = M3::float_t(2.0* (YResp[i] - YResp[i+1]) + dx * (Tangents[i] + Tangents[i+1]));
502
503 Par[i][0] = b / dx;
504 Par[i][1] = c / (dx * dx);
505 Par[i][2] = d / (dx * dx * dx);
506
507 if((Par[i][0] == -999) | (Par[i][1] == -999) | (Par[i][2] ==-999) | (Par[i][0] == -999.999) | (Par[i][1] == -999.999) | (Par[i][2] ==-999.999)){
508 MACH3LOG_INFO("Bad spline parameters for segment {}: (b, c, d) = {}, {}, {}. This will cause problems with GPU.",
509 i, Par[i][0], Par[i][1], Par[i][2]);
510 }
511 MACH3LOG_TRACE("b: {}", b);
512 MACH3LOG_TRACE("dx: {}, x_0: {}, x_1: {}", dx, XPos[i], XPos[i+1]);
513 MACH3LOG_TRACE(" y_0: {}, y_1: {}", YResp[i], YResp[i+1]);
514 }
515
516 // include params for final "segment" outside of the spline so that par array fits with x and y arrays,
517 // should never actually get used but if not set then the GPU code gets very angry
518 Par[nPoints-1][0] = 0.0;
519 Par[nPoints-1][1] = 0.0;
520 Par[nPoints-1][2] = 0.0;
521
522 // check the input spline for linear segments, if there are any then overwrite the calculated coefficients
523 // this will pretty much only ever be the case if they are set to be linear in samplePDFND i.e. the user wants it to be linear
524 for(int i = 0; i <nPoints-1; i++){
525 double x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
526 spline->GetCoeff(i, x, y, b, c, d);
527
528 if((c == 0.0 && d == 0.0)){
529 Par[i][0] = M3::float_t(b);
530 Par[i][1] = 0.0;
531 Par[i][2] = 0.0;
532 }
533 }
534 delete[] Secants;
535 delete[] Tangents;
536 } // end of if(nPoints !=2)
537 }
538 else
539 {
540 MACH3LOG_ERROR("Unsupported interpolation type {}", static_cast<int>(InterPolation));
541 throw MaCh3Exception(__FILE__ , __LINE__ );
542 }

545 spline = NULL;
546 }

566 inline int FindX(double x) {
567 // The segment we're interested in (klow in ROOT code)
568 int segment = 0;
569 int kHigh = nPoints-1;
570 // If the variation is below the lowest saved spline point
571 if (x <= XPos[0]){
572 segment = 0;
573 // If the variation is above the highest saved spline point
574 } else if (x >= XPos[nPoints-1]) {
575 // Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
576 segment = kHigh;
577 // If the variation is between the maximum and minimum, perform a binary search
578 } else {
579 // The top point we've got
580 int kHalf = 0;
581 // While there is still a difference in the points (we haven't yet found the segment)
582 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
583 while (kHigh - segment > 1) {
584 // Increment the half-step
585 kHalf = (segment + kHigh)/2;
586 // If our variation is above the kHalf, set the segment to kHalf
587 if (x > XPos[kHalf]) {
588 segment = kHalf;
589 // Else move kHigh down
590 } else {
591 kHigh = kHalf;
592 }
593 } // End the while: we've now done our binary search
594 } // End the else: we've now found our point
595 if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
596 return segment;
597 }

600 inline double Eval(double var) override {
601 // Get the segment for this variation
602 int segment = FindX(var);
603 // The get the coefficients for this variation
604 M3::float_t x = M3::float_t(-999.99), y = M3::float_t(-999.99), b = M3::float_t(-999.99), c = M3::float_t(-999.99), d = M3::float_t(-999.99);
605 GetCoeff(segment, x, y, b, c, d);

607 // Evaluate the third order polynomial

671// *****************************************
672 int Np = spl->GetNp();
673 M3::float_t x, y, b, c, d;
674 // Go through spline segment parameters,
675 // Get y values for each spline knot,
676 // Every knot must evaluate to 1.0 to create a flat spline

678 spl->GetCoeff(i, x, y, b, c, d);

681 }
682 }

692 std::vector<TSpline3*>::iterator InnerIt;

703 TempVector.reserve(OuterIt->size());
704 int InnerCounter = 0;
705 // Loop over each TSpline3 pointer
706 for (InnerIt = OuterIt->begin(); InnerIt != OuterIt->end(); ++InnerIt, ++InnerCounter) {
707 // Here's our delicious TSpline3 object

715 // Push back onto new vector
716 TempVector.push_back(red);
717 } // End inner for loop
718 ReducedVector.push_back(TempVector);
719 } // End outer for loop
