568 // helper function to read from the translation config file to get an option for a particular sub
569 // node (e.g. Parameters or Samples) returns true and sets "ret" if the option is specified for this
570 // fitter and parameter otherwise returns false
571 template <typename T>
572 inline bool getFitterSpecificOption(const std::string &fitter, const std::string &option, T &ret, const std::string &parameter,
573 YAML::Node subConfig) const{
574 if (subConfig[parameter])
575 {
576
577 // EM: this is config definition of fitter specific names for this parameter
578 YAML::Node paramTranslation = subConfig[parameter];
579
580 if (paramTranslation[fitter])
581 {
582 // EM: then this is definition of how to find parameter in the specified fitter
583 YAML::Node fitterParamTranslation = paramTranslation[fitter];
584
585 if (fitterParamTranslation[option])
586 {
587 ret = fitterParamTranslation[option].as<T>();
588 return true;
589 }
590 }
591 }
592 return false;
593 }
