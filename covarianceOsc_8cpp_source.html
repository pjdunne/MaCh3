69
70 covarianceBase::proposeStep();

72 // HW :: This method is a tad hacky but modular arithmetic gives me a headache.
73 // It should now automatically set dcp to be with [-pi, pi]
74 if(_fPropVal[kDeltaCP] > TMath::Pi()) {
75 _fPropVal[kDeltaCP] = -1*TMath::Pi() + std::fmod(_fPropVal[kDeltaCP], TMath::Pi());
76 } else if (_fPropVal[kDeltaCP] < -TMath::Pi()) {
77 _fPropVal[kDeltaCP] = TMath::Pi() + std::fmod(_fPropVal[kDeltaCP], TMath::Pi());
78 }
79
80 // Okay now we've done the standard steps, we can add in our nice flips
81 // hierarchy flip first
82 if(random_number[0]->Uniform() < 0.5 && flipdelM){
83 _fPropVal[kDeltaM23] *= -1;
84 }
85 // now octant flip
86 if(random_number[0]->Uniform() < 0.5) {
87 // flip octant around point of maximal disappearance (0.5112)
88 // this ensures we move to a parameter value which has the same oscillation probability
89 _fPropVal[kSinTheta23] = 0.5112 - (_fPropVal[kSinTheta23] - 0.5112);
