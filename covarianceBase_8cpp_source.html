524// *************************************
525// Throw each parameter within their 1 sigma range
526// Used to start the chain in different states
527void covarianceBase::RandomConfiguration() {
528// *************************************
529 // Have the 1 sigma for each parameter in each covariance class, sweet!
530 // Don't want to change the nominal array because that's what determines our likelihood
531 // Want to change the fParProp, fParCurr, fParInit
532 // fParInit and the others will already be set
533 for (int i = 0; i < _fNumPar; ++i) {
534 // Check if parameter is fixed first: if so don't randomly throw
535 if (isParameterFixed(i)) continue;
536 // Check that the sigma range is larger than the parameter range
537 // If not, throw in the valid parameter range instead
538 const double paramrange = _fUpBound[i] - _fLowBound[i];
539 const double sigma = sqrt((*covMatrix)(i,i));
540 double throwrange = sigma;
541 if (paramrange < sigma) throwrange = paramrange;
542
543 _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
544 // Try again if we the initial parameter proposal falls outside of the range of the parameter
545 // Really only relevant for the xsec parameters; the flux and ND280 have -999 and 999 set to the limits!
546 int throws = 0;
547 while (_fPropVal[i] > _fUpBound[i] || _fPropVal[i] < _fLowBound[i]) {
548 if (throws > 1000) {
549 MACH3LOG_WARN("Tried {} times to throw parameter {} but failed", throws, i);
550 MACH3LOG_WARN("Matrix: {}", matrixName);
551 MACH3LOG_WARN("Param: {}", _fNames[i]);
552 throw MaCh3Exception(__FILE__ , __LINE__ );
553 }
554 _fPropVal[i] = _fPreFitValue[i] + random_number[0]->Gaus(0, 1)*throwrange;
555 throws++;
556 }
557 MACH3LOG_INFO("Setting current step in {} param {} = {} from {}", matrixName, i, _fPropVal[i], _fCurrVal[i]);
558 _fCurrVal[i] = _fPropVal[i];
559 }
560 if (pca) TransferToPCA();
561}

612 randParams[i] = 0.0;
613 }
614 } // end for
615 // If we're in the PCA basis we instead throw parameters there (only _fNumParPCA parameter)
616 } else {
617 // Scale the random parameters by the sqrt of eigen values for the throw
618 #ifdef MULTITHREAD
619 #pragma omp parallel for
620 #endif
621 for (int i = 0; i < _fNumPar; ++i)
622 {
623 if (fParSigma_PCA[i] > 0. && i < _fNumParPCA)
624 {
625#ifdef MULTITHREAD
626 randParams[i] = random_number[omp_get_thread_num()]->Gaus(0,1);
627#else
628 randParams[i] = random_number[0]->Gaus(0,1);
629#endif
630 } else { // If parameter IS fixed or out od bounds
631 randParams[i] = 0.0;
632 }
633 }
634 }
635}

637// ************************************************
638// Correlate the steps by setting the proposed step of a parameter to its current value + some correlated throw
639void covarianceBase::CorrelateSteps() {
640// ************************************************
641 //KS: Using custom function compared to ROOT one with 8 threads we have almost factor 2 performance increase, by replacing TMatrix with just double we increase it even more
642 MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
643
644 // If not doing PCA
645 if (!pca) {
646 #ifdef MULTITHREAD
647 #pragma omp parallel for
648 #endif
649 for (int i = 0; i < _fNumPar; ++i) {
650 if (_fError[i] > 0.) {
651 _fPropVal[i] = _fCurrVal[i] + corr_throw[i]*_fGlobalStepScale*_fIndivStepScale[i];
652 }
653 }
654 // If doing PCA throw uncorrelated in PCA basis (orthogonal basis by definition)
655 } else {
656 // Throw around the current step
657 #ifdef MULTITHREAD
658 #pragma omp parallel for
659 #endif
660 for (int i = 0; i < _fNumParPCA; ++i)
661 {
662 if (fParSigma_PCA[i] > 0.)
663 {
664 double IndStepScale = 1.;
665 //KS: If undecomposed parameter apply individual step scale and Cholesky for better acceptance rate
666 if(isDecomposed_PCA[i] >= 0)
667 {
668 IndStepScale *= _fIndivStepScale[isDecomposed_PCA[i]];
669 IndStepScale *= corr_throw[isDecomposed_PCA[i]];
670 }
671 //If decomposed apply only random number
672 else
673 {
674 IndStepScale *= randParams[i];
675 //KS: All PCA-ed parameters have the same step scale
676 IndStepScale *= _fIndivStepScale[FirstPCAdpar];
677 }
678 fParProp_PCA(i) = fParCurr_PCA(i)+_fGlobalStepScale*IndStepScale*PCAObj.eigen_values_master[i];
679 }
680 }
681 // Then update the parameter basis
682 TransferToParam();
683 }
684}
685// ********************************************
686// Update so that current step becomes the previously proposed step
687void covarianceBase::acceptStep() {
688// ********************************************
689 if (!pca) {
690 #ifdef MULTITHREAD
691 #pragma omp parallel for
692 #endif
693 for (int i = 0; i < _fNumPar; ++i) {
694 // Update state so that current state is proposed state
695 _fCurrVal[i] = _fPropVal[i];
696 }
697 } else {
698 // Update the book-keeping for the output
699 #ifdef MULTITHREAD
700 #pragma omp parallel for
701 #endif
702 for (int i = 0; i < _fNumParPCA; ++i) {
703 fParCurr_PCA(i) = fParProp_PCA(i);
704 }
705 // Then update the parameter basis
706 TransferToParam();
707 }

710// Throw the proposed parameter by mag sigma
711// Should really just have the user specify this throw by having argument double
712void covarianceBase::throwParProp(const double mag) {
713// ********************************************
714 randomize();
715 if (!pca) {
716 // Make the correlated throw
717 MatrixVectorMulti(corr_throw, throwMatrixCholDecomp, randParams, _fNumPar);
718 // Number of sigmas we throw
719 for (int i = 0; i < _fNumPar; i++) {
720 if (_fError[i] > 0.)
721 _fPropVal[i] = _fCurrVal[i] + corr_throw[i]*mag;
722 }
723 } else {
724 for (int i = 0; i < _fNumPar; i++) {
725 fParProp_PCA(i) = fParCurr_PCA(i)+mag*randParams[i];
726 }
727 // And update the fParCurr in the basis
728 // Then update the fParProp in the parameter basis using the transfer matrix, so likelihood is evaluated correctly
729 TVectorD proposed = PCAObj.TransferMat*fParProp_PCA;
730 for (int i = 0; i < _fNumPar; ++i) {
731 if (fParSigma_PCA[i] > 0.) {
732 _fPropVal[i] = proposed(i);
733 }
734 }
735 }

796// Get the likelihood in the case where we want to include priors on the parameters
797// fParEvalLikelihood stores if we want to evaluate the likelihood for the given parameter
798// true = evaluate likelihood (so run with a prior)
799// false = don't evaluate likelihood (so run without a prior)
800double covarianceBase::CalcLikelihood() {
801// ********************************************
802 double logL = 0.0;
803 #ifdef MULTITHREAD
804 #pragma omp parallel for reduction(+:logL)
805 #endif
806 for(int i = 0; i < _fNumPar; ++i){
807 #ifdef MULTITHREAD
808 #pragma omp simd
809 #endif
810 for (int j = 0; j <= i; ++j) {
811 if (!_fFlatPrior[i] && !_fFlatPrior[j]) {
812 //KS: Since matrix is symmetric we can calculate non diagonal elements only once and multiply by 2, can bring up to factor speed decrease.
813 short int scale = (i != j) ? 2 : 1;
814 logL += scale * 0.5*(_fPropVal[i] - _fPreFitValue[i])*(_fPropVal[j] - _fPreFitValue[j])*InvertCovMatrix[i][j];
815 }
816 }
817 }
818 return logL;
819}

1086// ********************************************
1087//Makes sure that matrix is positive-definite by adding a small number to on-diagonal elements
1088void covarianceBase::MakePosDef(TMatrixDSym *cov) {
1089// ********************************************
1090 //DB Save original warning state and then increase it in this function to suppress 'matrix not positive definite' messages
1091 //Means we no longer need to overload
1092 if(cov == nullptr){
1093 cov = &*covMatrix;
1094 }
1095
1096 int originalErrorWarning = gErrorIgnoreLevel;
1097 gErrorIgnoreLevel = kFatal;
1098
1099 //DB Loop 1000 times adding 1e-9 which tops out at 1e-6 shift on the diagonal before throwing error
1100 int MaxAttempts = 1e5;
1101 int iAttempt = 0;
1102 bool CanDecomp = false;
1103 TDecompChol chdcmp;
1104
1105 for (iAttempt = 0; iAttempt < MaxAttempts; iAttempt++) {
1106 chdcmp = TDecompChol(*cov);
1107 if (chdcmp.Decompose()) {
1108 CanDecomp = true;
1109 break;
1110 } else {
1111 #ifdef MULTITHREAD
1112 #pragma omp parallel for
1113 #endif
1114 for (int iVar = 0 ; iVar < _fNumPar; iVar++) {
1115 (*cov)(iVar,iVar) += pow(10,-9);
1116 }
1117 }
1118 }
1119
1120 if (!CanDecomp) {
1121 MACH3LOG_ERROR("Tried {} times to shift diagonal but still can not decompose the matrix", MaxAttempts);
1122 MACH3LOG_ERROR("This indicates that something is wrong with the input matrix");
1123 throw MaCh3Exception(__FILE__ , __LINE__ );
1124 }
1125 if(AdaptiveHandler.total_steps < 2) {
1126 MACH3LOG_INFO("Had to shift diagonal {} time(s) to allow the covariance matrix to be decomposed", iAttempt);
1127 }
1128 //DB Resetting warning level
1129 gErrorIgnoreLevel = originalErrorWarning;
1130
1131 return;
1132}

1148// ********************************************

1176 #ifdef MULTITHREAD

1178 #endif
1179 for (int i = 0; i < _fNumPar; ++i)
