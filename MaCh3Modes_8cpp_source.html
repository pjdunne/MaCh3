95// *******************
96void MaCh3Modes::PrepareMap() {
97// *******************
98
99 int maxElement = 0;
100 for(int i = 0; i < NModes; ++i) {
101 for(size_t j = 0; j < fMode[i].GeneratorMaping.size(); j++) {
102 maxElement = std::max(maxElement, fMode[i].GeneratorMaping[j]);
103 }
104 }
105 ModeMap.resize(maxElement+1, Mode["UNKNOWN_BAD"]);
106
107 for(int m = 0; m < NModes; ++m) {
108 for(size_t i = 0; i < fMode[m].GeneratorMaping.size(); ++i) {
109 if(fMode[m].GeneratorMaping[i] < 0) {
110 MACH3LOG_ERROR("Negative value of Mode {} for mode {}", fMode[m].GeneratorMaping[i], fMode[m].Name);
111 throw MaCh3Exception(__FILE__ , __LINE__ );
112 }
113 if(ModeMap[fMode[m].GeneratorMaping[i]] != Mode["UNKNOWN_BAD"])
114 {
115 MACH3LOG_ERROR("Generator mode {} already defined in mode {} can't do this for mode {}", fMode[m].GeneratorMaping[i], fMode[ModeMap[fMode[m].GeneratorMaping[i]]].Name, fMode[m].Name);
116 throw MaCh3Exception(__FILE__ , __LINE__ );
117 }
118 ModeMap[fMode[m].GeneratorMaping[i]] = m;
119 }
120 }
121}
