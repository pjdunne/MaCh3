112inline std::string TMacroToString(const TMacro& macro) {
113// **********************
114 std::stringstream ss;
115
116 // Retrieve lines from TMacro
117 TList* linesList = macro.GetListOfLines();
118 if (!linesList) {
119 MACH3LOG_ERROR("Failed to retrieve lines from TMacro.");
120 return "";
121 }
122
123 TIter nextLine(linesList);
124 TObject *obj = nullptr;
125 while ((obj = nextLine())) {
126 TObjString* line = dynamic_cast<TObjString*>(obj);
127 if (!line) {
128 MACH3LOG_ERROR("Failed to cast object to TObjString.");
129 continue;
130 }
131 ss << line->GetString() << std::endl;
132 }
133
134 return ss.str();
135}

174inline bool compareYAMLNodes(const YAML::Node& node1, const YAML::Node& node2) {
175// **********************
176 // Check if the types of the nodes match
177 if (node1.Type() != node2.Type()) {
178 return false;
179 }
180
181 // Compare scalar types (like strings, numbers)
182 if (node1.IsScalar() && node2.IsScalar()) {
183 return node1.as<std::string>() == node2.as<std::string>();
184 }
185
186 // Compare sequences (like YAML lists)
187 if (node1.IsSequence() && node2.IsSequence()) {
188 if (node1.size() != node2.size()) {
189 return false;
190 }
191 for (std::size_t i = 0; i < node1.size(); ++i) {
192 if (!compareYAMLNodes(node1[i], node2[i])) {
193 return false;
194 }
195 }
196 return true;
197 }
198
199 // Compare maps (like YAML dictionaries)
200 if (node1.IsMap() && node2.IsMap()) {
201 if (node1.size() != node2.size()) {
202 return false;
203 }
204 for (auto it1 = node1.begin(); it1 != node1.end(); ++it1) {
205 auto key = it1->first.as<std::string>();
206 if (!node2[key] || !compareYAMLNodes(it1->second, node2[key])) {
207 return false;
208 }
209 }
210 return true;
211 }
212
213 // Default case: if it's neither scalar, sequence, nor map, consider it unequal
214 return false;
215}
