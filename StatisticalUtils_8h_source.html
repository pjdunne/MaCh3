197inline double GetBetaParameter(const double data, const double mc, const double w2, TestStatistic TestStat) {
198// ****************
199 double Beta = 0.0;
200
201 if (TestStat == kDembinskiAbdelmottele) {
202 //the so-called effective count
203 const double k = mc*mc / w2;
204 //Calculate beta which is scaling factor between true and generated MC
205 Beta = (data + k) / (mc + k);
206 }
207 //KS: Below is technically only true for Cowan's BB, which will not be true for Poisson or IceCube, because why not...
208 else {
209 // CW: Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
210 const double fractional = std::sqrt(w2)/mc;
211 // CW: -b/2a in quadratic equation
212 const double temp = mc*fractional*fractional-1;
213 // CW: b^2 - 4ac in quadratic equation
214 const double temp2 = temp*temp + 4*data*fractional*fractional;
215 if (temp2 < 0) {
216 MACH3LOG_ERROR("Negative square root in Barlow Beeston coefficient calculation!");
217 throw MaCh3Exception(__FILE__ , __LINE__ );
218 }
219 // CW: Solve for the positive beta
220 Beta = (-1*temp+std::sqrt(temp2))/2.;
221 }
222 return Beta;
223}

229inline double GetSubOptimality(const std::vector<double>& EigenValues, const int TotalTarameters) {
230// *********************
231 double sum_eigenvalues_squared_inv = 0.0;
232 double sum_eigenvalues_inv = 0.0;
233 for (unsigned int j = 0; j < EigenValues.size(); j++)
234 {
235 //KS: IF Eigen values are super small skip them
236 //if(EigenValues[j] < 0.0000001) continue;
237 sum_eigenvalues_squared_inv += std::pow(EigenValues[j], -2);
238 sum_eigenvalues_inv += 1.0 / EigenValues[j];
239 }
240 const double SubOptimality = TotalTarameters * sum_eigenvalues_squared_inv / std::pow(sum_eigenvalues_inv, 2);
241 return SubOptimality;
242}

289inline void GetHPD(TH1D* const hist, double& Mean, double& Error, double& Error_p, double& Error_m, const double coverage = 0.6827) {
290// ****************
291 // Get the bin which has the largest posterior density
292 const int MaxBin = hist->GetMaximumBin();
293 // And it's value
294 const double peakval = hist->GetBinCenter(MaxBin);
295
296 // The total integral of the posterior
297 const long double Integral = hist->Integral();
298 //KS: and integral of left handed and right handed parts
299 const long double LowIntegral = hist->Integral(1, MaxBin-1) + hist->GetBinContent(MaxBin)/2.0;
300 const long double HighIntegral = hist->Integral(MaxBin+1, hist->GetNbinsX()) + hist->GetBinContent(MaxBin)/2.0;
301
302 // Keep count of how much area we're covering
303 //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
304 long double sum = hist->GetBinContent(MaxBin)/2.0;
305
306 // Counter for current bin
307 int CurrBin = MaxBin;
308 while (sum/HighIntegral < coverage && CurrBin < hist->GetNbinsX()) {
309 CurrBin++;
310 sum += hist->GetBinContent(CurrBin);
311 }
312 const double sigma_p = std::fabs(hist->GetBinCenter(MaxBin)-hist->GetXaxis()->GetBinUpEdge(CurrBin));
313 // Reset the sum
314 //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
315 sum = hist->GetBinContent(MaxBin)/2.0;
316
317 // Reset the bin counter
318 CurrBin = MaxBin;
319 // Counter for current bin
320 while (sum/LowIntegral < coverage && CurrBin > 1) {
321 CurrBin--;
322 sum += hist->GetBinContent(CurrBin);
323 }
324 const double sigma_m = std::fabs(hist->GetBinCenter(CurrBin)-hist->GetBinLowEdge(MaxBin));
325
326 // Now do the double sided HPD
327 //KS: Start sum from the HPD
328 sum = hist->GetBinContent(MaxBin);
329 int LowBin = MaxBin;
330 int HighBin = MaxBin;
331 long double LowCon = 0.0;
332 long double HighCon = 0.0;
333
334 while (sum/Integral < coverage && (LowBin > 0 || HighBin < hist->GetNbinsX()+1))
335 {
336 LowCon = 0.0;
337 HighCon = 0.0;
338 //KS:: Move further only if you haven't reached histogram end
339 if(LowBin > 1)
340 {
341 LowBin--;
342 LowCon = hist->GetBinContent(LowBin);
343 }
344 if(HighBin < hist->GetNbinsX())
345 {
346 HighBin++;
347 HighCon = hist->GetBinContent(HighBin);
348 }
349
350 // If we're on the last slice and the lower contour is larger than the upper
351 if ((sum+LowCon+HighCon)/Integral > coverage && LowCon > HighCon) {
352 sum += LowCon;
353 break;
354 // If we're on the last slice and the upper contour is larger than the lower
355 } else if ((sum+LowCon+HighCon)/Integral > coverage && HighCon >= LowCon) {
356 sum += HighCon;
357 break;
358 } else {
359 sum += LowCon + HighCon;
360 }
361 }
362
363 double sigma_hpd = 0.0;
364 if (LowCon > HighCon) {
365 sigma_hpd = std::fabs(hist->GetBinLowEdge(LowBin)-hist->GetBinCenter(MaxBin));
366 } else {
367 sigma_hpd = std::fabs(hist->GetXaxis()->GetBinUpEdge(HighBin)-hist->GetBinCenter(MaxBin));
368 }
369
370 Mean = peakval;
371 Error = sigma_hpd;
372 Error_p = sigma_p;
373 Error_m = sigma_m;
374}

382inline void GetCredibleInterval(TH1D* const hist, TH1D* hpost_copy, const double coverage = 0.6827) {
383// ***************
384 if(coverage > 1)
385 {
386 MACH3LOG_ERROR("Specified Credible Interval is greater that 1 and equal to {} Should be between 0 and 1", coverage);
387 throw MaCh3Exception(__FILE__ , __LINE__ );
388 }
389 //KS: Reset first copy of histogram
390 hpost_copy->Reset("");
391 hpost_copy->Fill(0.0, 0.0);
392
393 //KS: Temporary structure to be thread save
394 double *hist_copy = new double[hist->GetXaxis()->GetNbins()+1];
395 bool *hist_copy_fill = new bool[hist->GetXaxis()->GetNbins()+1];
396 for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
397 {
398 hist_copy[i] = hist->GetBinContent(i);
399 hist_copy_fill[i] = false;
400 }
401
403 const long double Integral = hist->Integral();
404 long double sum = 0;
405
406 while ((sum / Integral) < coverage)
407 {
409 int max_entry_bin = 0;
410 double max_entries = 0.;
411 for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
412 {
413 if (hist_copy[i] > max_entries)
414 {
415 max_entries = hist_copy[i];
416 max_entry_bin = i;
417 }
418 }
420 hist_copy[max_entry_bin] = -1.;
421 hist_copy_fill[max_entry_bin] = true;
422
423 sum += max_entries;
424 }
425 //KS: Now fill our copy only for bins which got included in coverage region
426 for(int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
427 {
428 if(hist_copy_fill[i]) hpost_copy->SetBinContent(i, hist->GetBinContent(i));
429 }
430
431 delete[] hist_copy;
432 delete[] hist_copy_fill;
433
434 return;
435}

467 {

470 int max_entry_bin_y = 0;

472 for (int i = 0; i <= hist2D->GetXaxis()->GetNbins(); ++i)
473 {
474 for (int j = 0; j <= hist2D->GetYaxis()->GetNbins(); ++j)
475 {
476 if (hist_copy[i][j] > max_entries)
477 {
478 max_entries = hist_copy[i][j];
479 max_entry_bin_x = i;
480 max_entry_bin_y = j;
481 }
482 }
483 }
485 hist_copy[max_entry_bin_x][max_entry_bin_y] = -1.;
486
487 sum += max_entries;

491
492 //Delete temporary arrays
493 for (int i = 0; i <= hist2D->GetXaxis()->GetNbins(); ++i)
