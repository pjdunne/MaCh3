116inline void CheckBonferoniCorrectedpValue(const std::vector<std::string>& SampleNameVec,
117 const std::vector<double>& PValVec,
118 const double Threshold = 0.05) {
119// ****************
120 MACH3LOG_INFO("");
121 if(SampleNameVec.size() != PValVec.size())
122 {
123 MACH3LOG_ERROR("Size of vectors do not match");
124 throw MaCh3Exception(__FILE__ , __LINE__ );
125 }
126 const size_t NumberOfStatisticalTests = SampleNameVec.size();
127 //KS: 0.05 or 5% is value used by T2K.
128 const double StatisticalSignificanceDown = Threshold / double(NumberOfStatisticalTests);
129 const double StatisticalSignificanceUp = 1 - StatisticalSignificanceDown;
130 MACH3LOG_INFO("Bonferroni-corrected statistical significance level: {:.2f}", StatisticalSignificanceDown);
131
132 int Counter = 0;
133 for(unsigned int i = 0; i < SampleNameVec.size(); i++)
134 {
135 if( (PValVec[i] < 0.5 && PValVec[i] < StatisticalSignificanceDown) )
136 {
137 MACH3LOG_INFO("Sample {} indicates disagreement between the model predictions and the data", SampleNameVec[i]);
138 MACH3LOG_INFO("Bonferroni-corrected statistical significance level: {:.2f} p-value: {:.2f}", StatisticalSignificanceDown, PValVec[i]);
139 Counter++;
140 }
141 else if( (PValVec[i] > 0.5 && PValVec[i] > StatisticalSignificanceUp) )
142 {
143 MACH3LOG_INFO("Sample {} indicates disagreement between the model predictions and the data", SampleNameVec[i]);
144 MACH3LOG_INFO("Bonferroni-corrected statistical significance level: {:.2f} p-value: {:.2f}", StatisticalSignificanceUp, PValVec[i]);
145 Counter++;
146 }
147 }
148 if(Counter == 0) {
149 MACH3LOG_INFO("Every sample passed Bonferroni-corrected statistical significance level test");
150 } else {
151 MACH3LOG_WARN("{} samples didn't pass Bonferroni-corrected statistical significance level test", Counter);
152 }
153 MACH3LOG_INFO("");
154}

192inline double GetBetaParameter(const double data, const double mc, const double w2, TestStatistic TestStat) {
193// ****************
194 double Beta = 0.0;
195
196 if (TestStat == kDembinskiAbdelmottele) {
197 //the so-called effective count
198 const double k = mc*mc / w2;
199 //Calculate beta which is scaling factor between true and generated MC
200 Beta = (data + k) / (mc + k);
201 }
202 //KS: Below is technically only true for Cowan's BB, which will not be true for Poisson or IceCube, because why not...
203 else {
204 // CW: Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
205 const double fractional = std::sqrt(w2)/mc;
206 // CW: -b/2a in quadratic equation
207 const double temp = mc*fractional*fractional-1;
208 // CW: b^2 - 4ac in quadratic equation
209 const double temp2 = temp*temp + 4*data*fractional*fractional;
210 if (temp2 < 0) {
211 MACH3LOG_ERROR("Negative square root in Barlow Beeston coefficient calculation!");
212 throw MaCh3Exception(__FILE__ , __LINE__ );
213 }
214 // CW: Solve for the positive beta
215 Beta = (-1*temp+std::sqrt(temp2))/2.;
216 }
217 return Beta;
218}

224inline double GetSubOptimality(const std::vector<double>& EigenValues, const int TotalTarameters) {
225// *********************
226 double sum_eigenvalues_squared_inv = 0.0;
227 double sum_eigenvalues_inv = 0.0;
228 for (unsigned int j = 0; j < EigenValues.size(); j++)
229 {
230 //KS: IF Eigen values are super small skip them
231 //if(EigenValues[j] < 0.0000001) continue;
232 sum_eigenvalues_squared_inv += std::pow(EigenValues[j], -2);
233 sum_eigenvalues_inv += 1.0 / EigenValues[j];
234 }
235 const double SubOptimality = TotalTarameters * sum_eigenvalues_squared_inv / std::pow(sum_eigenvalues_inv, 2);
236 return SubOptimality;
237}

284inline void GetHPD(TH1D* const hist, double& Mean, double& Error, double& Error_p, double& Error_m, const double coverage = 0.6827) {
285// ****************
286 // Get the bin which has the largest posterior density
287 const int MaxBin = hist->GetMaximumBin();
288 // And it's value
289 const double peakval = hist->GetBinCenter(MaxBin);
290
291 // The total integral of the posterior
292 const long double Integral = hist->Integral();
293 //KS: and integral of left handed and right handed parts
294 const long double LowIntegral = hist->Integral(1, MaxBin-1) + hist->GetBinContent(MaxBin)/2.0;
295 const long double HighIntegral = hist->Integral(MaxBin+1, hist->GetNbinsX()) + hist->GetBinContent(MaxBin)/2.0;
296
297 // Keep count of how much area we're covering
298 //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
299 long double sum = hist->GetBinContent(MaxBin)/2.0;
300
301 // Counter for current bin
302 int CurrBin = MaxBin;
303 while (sum/HighIntegral < coverage && CurrBin < hist->GetNbinsX()) {
304 CurrBin++;
305 sum += hist->GetBinContent(CurrBin);
306 }
307 const double sigma_p = std::fabs(hist->GetBinCenter(MaxBin)-hist->GetXaxis()->GetBinUpEdge(CurrBin));
308 // Reset the sum
309 //KS: Take only half content of HPD bin as one half goes for right handed error and the other for left handed error
310 sum = hist->GetBinContent(MaxBin)/2.0;
311
312 // Reset the bin counter
313 CurrBin = MaxBin;
314 // Counter for current bin
315 while (sum/LowIntegral < coverage && CurrBin > 1) {
316 CurrBin--;
317 sum += hist->GetBinContent(CurrBin);
318 }
319 const double sigma_m = std::fabs(hist->GetBinCenter(CurrBin)-hist->GetBinLowEdge(MaxBin));
320
321 // Now do the double sided HPD
322 //KS: Start sum from the HPD
323 sum = hist->GetBinContent(MaxBin);
324 int LowBin = MaxBin;
325 int HighBin = MaxBin;
326 long double LowCon = 0.0;
327 long double HighCon = 0.0;
328
329 while (sum/Integral < coverage && (LowBin > 0 || HighBin < hist->GetNbinsX()+1))
330 {
331 LowCon = 0.0;
332 HighCon = 0.0;
333 //KS:: Move further only if you haven't reached histogram end
334 if(LowBin > 1)
335 {
336 LowBin--;
337 LowCon = hist->GetBinContent(LowBin);
338 }
339 if(HighBin < hist->GetNbinsX())
340 {
341 HighBin++;
342 HighCon = hist->GetBinContent(HighBin);
343 }
344
345 // If we're on the last slice and the lower contour is larger than the upper
346 if ((sum+LowCon+HighCon)/Integral > coverage && LowCon > HighCon) {
347 sum += LowCon;
348 break;
349 // If we're on the last slice and the upper contour is larger than the lower
350 } else if ((sum+LowCon+HighCon)/Integral > coverage && HighCon >= LowCon) {
351 sum += HighCon;
352 break;
353 } else {
354 sum += LowCon + HighCon;
355 }
356 }
357
358 double sigma_hpd = 0.0;
359 if (LowCon > HighCon) {
360 sigma_hpd = std::fabs(hist->GetBinLowEdge(LowBin)-hist->GetBinCenter(MaxBin));
361 } else {
362 sigma_hpd = std::fabs(hist->GetXaxis()->GetBinUpEdge(HighBin)-hist->GetBinCenter(MaxBin));
363 }
364
365 Mean = peakval;
366 Error = sigma_hpd;
367 Error_p = sigma_p;
368 Error_m = sigma_m;
369}

375inline void GetCredibleInterval(TH1D* const hist, TH1D* hpost_copy, const double coverage = 0.6827) {
376// ***************
377 if(coverage > 1)
378 {
379 MACH3LOG_ERROR("Specified Credible Interval is greater that 1 and equal to {} Should be between 0 and 1", coverage);
380 throw MaCh3Exception(__FILE__ , __LINE__ );
381 }
382 //KS: Reset first copy of histogram
383 hpost_copy->Reset("");
384 hpost_copy->Fill(0.0, 0.0);
385
386 //KS: Temporary structure to be thread save
387 std::vector<double> hist_copy(hist->GetXaxis()->GetNbins()+1);
388 std::vector<bool> hist_copy_fill(hist->GetXaxis()->GetNbins()+1);
389 for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
390 {
391 hist_copy[i] = hist->GetBinContent(i);
392 hist_copy_fill[i] = false;
393 }
394
396 const long double Integral = hist->Integral();
397 long double sum = 0;
398
399 while ((sum / Integral) < coverage)
400 {
402 int max_entry_bin = 0;
403 double max_entries = 0.;
404 for (int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
405 {
406 if (hist_copy[i] > max_entries)
407 {
408 max_entries = hist_copy[i];
409 max_entry_bin = i;
410 }
411 }
413 hist_copy[max_entry_bin] = -1.;
414 hist_copy_fill[max_entry_bin] = true;
415
416 sum += max_entries;
417 }
418 //KS: Now fill our copy only for bins which got included in coverage region
419 for(int i = 0; i <= hist->GetXaxis()->GetNbins(); ++i)
420 {
421 if(hist_copy_fill[i]) hpost_copy->SetBinContent(i, hist->GetBinContent(i));
422 }
423}

448// ***************
449 if(coverage > 1)

452 throw MaCh3Exception(__FILE__ , __LINE__ );

454
455 //KS: Temporary structure to be thread save
456 std::vector<std::vector<double>> hist_copy(hist2D->GetXaxis()->GetNbins()+1,

459 for (int j = 0; j <= hist2D->GetYaxis()->GetNbins(); ++j) {

462 }
463
465 const long double Integral = hist2D->Integral();
466 long double sum = 0;
467
468 //We need to as ROOT requires array to set to contour
469 double Contour[1];
470 while ((sum / Integral) < coverage)
471 {
473 int max_entry_bin_x = 0;
474 int max_entry_bin_y = 0;
475 double max_entries = 0.;
476 for (int i = 0; i <= hist2D->GetXaxis()->GetNbins(); ++i)
477 {
478 for (int j = 0; j <= hist2D->GetYaxis()->GetNbins(); ++j)
479 {
480 if (hist_copy[i][j] > max_entries)
481 {
482 max_entries = hist_copy[i][j];
483 max_entry_bin_x = i;
484 max_entry_bin_y = j;
485 }
486 }
487 }
489 hist_copy[max_entry_bin_x][max_entry_bin_y] = -1.;
490
491 sum += max_entries;
492 Contour[0] = max_entries;
493 }
494 hist2D->SetContour(1, Contour);

533inline double ComputeKLDivergence(TH2Poly* DataPoly, TH2Poly* PolyMC) {
534// *********************
535 double klDivergence = 0.0;
536 double DataIntegral = NoOverflowIntegral(DataPoly);
537 double MCIntegral = NoOverflowIntegral(PolyMC);
538 for (int i = 1; i < DataPoly->GetNumberOfBins()+1; ++i)
539 {
540 if (DataPoly->GetBinContent(i) > 0 && PolyMC->GetBinContent(i) > 0) {
541 klDivergence += DataPoly->GetBinContent(i) / DataIntegral *
542 std::log((DataPoly->GetBinContent(i) / DataIntegral) / ( PolyMC->GetBinContent(i) / MCIntegral));
543 }
544 }
545 return klDivergence;
546}

552inline double FisherCombinedPValue(const std::vector<double>& pvalues) {
553// ********************
554
555 double testStatistic = 0;
556 for(size_t i = 0; i < pvalues.size(); i++)
557 {
558 const double pval = std::max(0.00001, pvalues[i]);
559 testStatistic += -2.0 * std::log(pval);
560 }
561 // Degrees of freedom is twice the number of p-values
562 int degreesOfFreedom = int(2 * pvalues.size());
563 double pValue = TMath::Prob(testStatistic, degreesOfFreedom);
564
565 return pValue;
566}

576// ********************

579 int ret = system(("cp " + FilePath + " " + TempFilePath).c_str());
580 if (ret != 0) {
581 MACH3LOG_WARN("Error: system call to copy file failed with code {}", ret);
582 }
583
584 TFile *inFile = TFile::Open(TempFilePath.c_str(), "UPDATE");
585 if (!inFile || inFile->IsZombie()) {
586 MACH3LOG_ERROR("Error opening file: {}", TempFilePath);
587 throw MaCh3Exception(__FILE__, __LINE__);
588 }
589
590 TTree *inTree = inFile->Get<TTree>("posteriors");
591 if (!inTree) {
592 MACH3LOG_ERROR("Error: TTree 'posteriors' not found in file.");
593 inFile->ls();
594 inFile->Close();
595 throw MaCh3Exception(__FILE__, __LINE__);
596 }
597
598 // Clone the structure without data
599 TTree *outTree = inTree->CloneTree(0);
600
601 // Loop over entries and apply thinning
602 Long64_t nEntries = inTree->GetEntries();
603 double retainedPercentage = (double(nEntries) / ThinningCut) / double(nEntries) * 100;
604 MACH3LOG_INFO("Thinning will retain {:.2f}% of chains", retainedPercentage);
605 for (Long64_t i = 0; i < nEntries; i++) {
606 if (i % (nEntries/10) == 0) {
607 MaCh3Utils::PrintProgressBar(i, nEntries);
608 }
609 if (i % ThinningCut == 0) {
610 inTree->GetEntry(i);
611 outTree->Fill();
612 }
613 }
614 inFile->WriteTObject(outTree, "posteriors", "kOverwrite");
615 inFile->Close();
616 delete inFile;
617
618 MACH3LOG_INFO("Thinned TTree saved and overwrote original in: {}", TempFilePath);
619}
