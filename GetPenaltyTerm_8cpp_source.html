42void ReadXSecFile(const std::string& inputFile)
43{
44 // Now read the MCMC file
45 TFile *TempFile = new TFile(inputFile.c_str(), "open");
46
47 // Get the matrix
48 TMatrixDSym *XSecMatrix = TempFile->Get<TMatrixDSym>("CovarianceFolder/xsec_cov");
49
50 // Get the settings for the MCMC
51 TMacro *Config = TempFile->Get<TMacro>("MaCh3_Config");
52 if (Config == nullptr) {
53 MACH3LOG_ERROR("Didn't find MaCh3_Config tree in MCMC file! {}", inputFile);
54 TempFile->ls();
55 throw MaCh3Exception(__FILE__ , __LINE__ );
56 }
57
58 YAML::Node Settings = TMacroToYAML(*Config);
59
60 //CW: Get the xsec Covariance matrix
61 std::vector<std::string> XsecCovPos = GetFromManager<std::vector<std::string>>(Settings["General"]["Systematics"]["XsecCovFile"], {"none"});
62 if(XsecCovPos.back() == "none")
63 {
64 MACH3LOG_WARN("Couldn't find XsecCov branch in output");
65 MaCh3Utils::PrintConfig(Settings);
66 throw MaCh3Exception(__FILE__ , __LINE__ );
67 }
68
69 //KS:Most inputs are in ${MACH3}/inputs/blarb.root
70 if (std::getenv("MACH3") != nullptr) {
71 MACH3LOG_INFO("Found MACH3 environment variable: {}", std::getenv("MACH3"));
72 for(unsigned int i = 0; i < XsecCovPos.size(); i++)
73 XsecCovPos[i].insert(0, std::string(std::getenv("MACH3"))+"/");
74 }
75
76 YAML::Node XSecFile;
77 XSecFile["Systematics"] = YAML::Node(YAML::NodeType::Sequence);
78 for(unsigned int i = 0; i < XsecCovPos.size(); i++)
79 {
80 YAML::Node YAMLDocTemp = YAML::LoadFile(XsecCovPos[i]);
81 for (const auto& item : YAMLDocTemp["Systematics"]) {
82 XSecFile["Systematics"].push_back(item);
83 }
84 }
85
86 size = XSecMatrix->GetNrows();
87
88 auto systematics = XSecFile["Systematics"];
89 for (auto it = systematics.begin(); it != systematics.end(); ++it)
90 {
91 auto const &param = *it;
92
93 ParamNames.push_back(param["Systematic"]["Names"]["FancyName"].as<std::string>());
94 nominal.push_back( param["Systematic"]["ParameterValues"]["PreFitValue"].as<double>() );
95
96 bool flat = false;
97 if (param["Systematic"]["FlatPrior"]) { flat = param["Systematic"]["FlatPrior"].as<bool>(); }
98 isFlat.push_back( flat );
99 }
100
101 XSecMatrix->Invert();
102 //KS: Let's use double as it is faster than TMatrix
103 invCovMatrix = new double*[size];
104 for (int i = 0; i < size; i++)
105 {
106 invCovMatrix[i] = new double[size];
107 for (int j = 0; j < size; ++j)
108 {
109 invCovMatrix[i][j] = -999;
110 }
111 }
112 #ifdef MULTITHREAD
113 #pragma omp parallel for collapse(2)
114 #endif
115 for (int i = 0; i < size; i++)
116 {
117 for (int j = 0; j < size; ++j)
118 {
119 invCovMatrix[i][j] = (*XSecMatrix)(i,j);
120 }
121 }
122
123 TempFile->Close();
124 delete TempFile;
125}

174 Chain->SetBranchStatus("*", false);

177 // Turn on the branches which we want for parameters
178 for (int i = 0; i < RelevantBranches; ++i)

202 FancyTittle.push_back(Set[2].as<std::string>());
203 }
204
205 const int NSets = int(SetsNames.size());
206
207 isRelevantParam.resize(NSets);
208 //Loop over sets in the config
209 for(int i = 0; i < NSets; i++)
210 {
211 isRelevantParam[i].resize(size);
212 int counter = 0;
213 //Loop over parameters in the Covariance object
214 for (int j = 0; j < size; j++)
215 {
216 isRelevantParam[i][j] = false;
217
218 //KS: Here we loop over all names and if parameters wasn't matched then we set it is relevant. For xsec it is easier to do it like this
219 if(Exclude[i])
220 {
221 bool found = false;
222 for (unsigned int k = 0; k < RemoveNames[i].size(); k++)
223 {
224 if (ParamNames[j].rfind(RemoveNames[i][k], 0) == 0)

226 found = true;
227 }
228 }
229 if(!found)

231 isRelevantParam[i][j] = true;
232 counter++;
233 }
234 }
235 //KS: Here is much simpler, if parameter matched then it is relevant
236 else
237 {
238 for (unsigned int k = 0; k < RemoveNames[i].size(); k++)
239 {
240 if (ParamNames[j].rfind(RemoveNames[i][k], 0) == 0)
241 {
242 isRelevantParam[i][j] = true;

244 break;

247 }
248 }
