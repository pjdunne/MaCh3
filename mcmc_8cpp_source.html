61// **********************
62// Do we accept the proposed step for all the parameters?
63void mcmc::CheckStep() {
64// **********************
65
66 bool accept = false;
67
68 // Set the acceptance probability to zero
69 accProb = 0.0;
70
71 // Calculate acceptance probability
72 if (anneal) accProb = TMath::Min(1.,TMath::Exp( -(logLProp-logLCurr) / (TMath::Exp(-step/AnnealTemp))));
73 else accProb = TMath::Min(1., TMath::Exp(logLCurr-logLProp));
74
75 // Get the random number
76 double fRandom = random->Rndm();
77
78 // Do the accept/reject
79 if (fRandom <= accProb) {
80 accept = true;
81 ++accCount;
82 } else {
83 accept = false;
84 }
85
86 #ifdef DEBUG
87 if (debug) debugFile << " logLProp: " << logLProp << " logLCurr: " << logLCurr << " accProb: " << accProb << " fRandom: " << fRandom << std::endl;
88 #endif
89
90 // Update all the handlers to accept the step
91 if (accept && !reject) {
92 logLCurr = logLProp;
93
94 if (osc) {
95 osc->acceptStep();
96 }
97
98 // Loop over systematics and accept
99 for (size_t s = 0; s < systematics.size(); ++s) {
100 systematics[s]->acceptStep();
101 }
102 }
103
104 stepClock->Stop();
105 stepTime = stepClock->RealTime();
106
107 // Write step to output tree
108 outTree->Fill();
109}

111// *******************
112// Run the Markov chain with all the systematic objects added
113void mcmc::runMCMC() {
114 // *******************
115
116 // Save the settings into the output file
117 SaveSettings();
118
119 // Prepare the output branches
120 PrepareOutput();
121
122 // Reconfigure the samples, systematics and oscillation for first weight
123 // ProposeStep sets logLProp
124 ProposeStep();
125 // Set the current logL to the proposed logL for the 0th step
126 // Accept the first step to set logLCurr: this shouldn't affect the MCMC because we ignore the first N steps in burn-in
127 logLCurr = logLProp;
128
129 // Begin MCMC
130 for (step = stepStart; step < stepStart+chainLength; ++step)
131 {
132 stepClock->Start();
133 // Set the initial rejection to false
134 reject = false;
135
136 // Print 10 steps in total
137 if ( (step-stepStart) % (chainLength/10) == 0) {
138 PrintProgress();
139 }
140
141 // Propose current step variation and save the systematic likelihood that results in this step being taken
142 // Updates logLProp
143 ProposeStep();
144
145 // Does the MCMC accept this step?
146 CheckStep();
147
148 // Auto save the output
149 if (step % auto_save == 0) outTree->AutoSave();
150 }
151
152 // Save all the MCMC output
153 SaveOutput();
154
155 // Process MCMC
156 ProcessMCMC();
157}

159// *******************
160// Do the initial reconfigure of the MCMC
161void mcmc::ProposeStep() {
162// *******************
163
164 // Initial likelihood
165 double llh = 0.0;
166
167 // Initiate to false
168 reject = false;
169
170 // Propose steps for the oscillation handlers
171 if (osc) {
172 osc->proposeStep();
173
174 // Now get the likelihoods for the oscillation
175 osc_llh = osc->GetLikelihood();
176
177 // Add the oscillation likelihoods to the reconfigure likelihoods
178 llh += osc_llh;
179
180 #ifdef DEBUG
181 if (debug) debugFile << "LLH for oscillation handler: " << llh << std::endl;
182 #endif
183 }
184
185 // Loop over the systematics and propose the initial step
186 for (size_t s = 0; s < systematics.size(); ++s) {
187 // Could throw the initial value here to do MCMC stability studies
188 // Propose the steps for the systematics
189 systematics[s]->proposeStep();
190
191 // Get the likelihood from the systematics
192 syst_llh[s] = systematics[s]->GetLikelihood();
193 llh += syst_llh[s];
194
195 #ifdef DEBUG
196 if (debug) debugFile << "LLH after " << systematics[s]->getName() << " " << llh << std::endl;
197 #endif
198 }
199
200 // Check if we've hit a boundary in the systematics
201 // In this case we can save time by not having to reconfigure the simulation
202 if (llh >= _LARGE_LOGL_) {
203 reject = true;
204 #ifdef DEBUG
205 if (debug) debugFile << "Rejecting based on boundary" << std::endl;
206 #endif
207 }
208
209 // Only reweight when we have a good parameter configuration
210 // This speeds things up considerably because for every bad parameter configuration we don't have to reweight the MC
211 if (!reject)
212 {
213 // Could multi-thread this
214 // But since sample reweight is multi-threaded it's probably better to do that
215 for (size_t i = 0; i < samples.size(); ++i)
216 {
217 samples[i]->reweight();
218 }
219
220 //DB for atmospheric event by event sample migration, need to fully reweight all samples to allow event passing prior to likelihood evaluation
221 for (size_t i = 0; i < samples.size(); ++i) {
222 // Get the sample likelihoods and add them
223 sample_llh[i] = samples[i]->GetLikelihood();
224 llh += sample_llh[i];
225 #ifdef DEBUG
226 if (debug) debugFile << "LLH after sample " << i << " " << llh << std::endl;
227 #endif
228 }
229
230 // For when we don't have to reweight, set sample to madness
231 } else {
232 for (size_t i = 0; i < samples.size(); ++i) {
233 // Set the sample_llh[i] to be madly high also to signify a step out of bounds
234 sample_llh[i] = _LARGE_LOGL_;
235 #ifdef DEBUG
236 if (debug) debugFile << "LLH after REJECT sample " << i << " " << llh << std::endl;
237 #endif
238 }
239 }
240
241 // Save the proposed likelihood (class member)
242 logLProp = llh;
243}
