235double GetMinimumInRange(TH1D *hist, double minRange, double maxRange)
236{
237 double MinVale = 1234567890.;
238 TAxis* xaxis = hist->GetXaxis();
239 for(int x = 1; x <= hist->GetNbinsX(); x++)
240 {
241 if(xaxis->GetBinLowEdge(x)>minRange && xaxis->GetBinUpEdge(x)<maxRange){
242 if(MinVale > hist->GetBinContent(x))MinVale = hist->GetBinContent(x);
243 }
244 }
245 return MinVale;
