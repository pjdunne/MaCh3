412void SampleSummary::AddThrow(std::vector<TH2Poly*> &SampleVector, std::vector<TH2Poly*> &W2Vec, const double LLHPenalty, const double Weight, const int DrawNumber) {
413// *******************
414
415 nThrows++;
416 //KS: Only make sense for PosteriorPredictive
417 if( !isPriorPredictive )RandomHist->Fill(DrawNumber);
418
419 const int size = int(SampleVector.size());
420 if (!CheckSamples(size)) throw MaCh3Exception(__FILE__ , __LINE__ );
421
422 // Push back the throw
423 MCVector.push_back(SampleVector);
424 LLHPenaltyVector.push_back(LLHPenalty);
425 WeightVector.push_back(Weight);
426 W2MCVector.push_back(W2Vec);
427
428 // Initialise the posterior hist
429 if (first_pass)
430 {
431 for (int SampleNum = 0; SampleNum < nSamples; ++SampleNum)
432 {
433 const int nXBins = 500;
434 //Initialise TH1D which corresponds to each bin in the sample's th2poly
435 std::string name = std::string(SampleVector[SampleNum]->GetName());
436 for (int i = 1; i <= maxBins[SampleNum]; ++i)
437 {
438 //Get PolyBin
439 TH2PolyBin* bin = static_cast<TH2PolyBin*>(SampleVector[SampleNum]->GetBins()->At(i-1));
440
441 // Just make a little fancy name
442 std::stringstream ss2;
443 ss2 << name << "_";
444 ss2 << "p_{#mu} (" << bin->GetXMin() << "-" << bin->GetXMax() << ")";
445 ss2 << " cos#theta_{#mu} (" << bin->GetYMin() << "-" << bin->GetYMax() << ")";
446
447 PosteriorHist[SampleNum][i] = std::make_unique<TH1D>(ss2.str().c_str(), ss2.str().c_str(),nXBins, 1, -1);
448 PosteriorHist[SampleNum][i]->SetDirectory(nullptr);
449 w2Hist[SampleNum][i] = std::make_unique<TH1D>(("w2_"+ss2.str()).c_str(), ("w2_"+ss2.str()).c_str(),nXBins, 1, -1);
450 w2Hist[SampleNum][i]->SetDirectory(nullptr);
451 if(DoBetaParam)
452 {
453 std::string betaName = "#beta_param_";
454 BetaHist[SampleNum][i] = std::make_unique<TH1D>((betaName + ss2.str()).c_str(), (betaName + ss2.str()).c_str(), 70, 1, -1);
455 BetaHist[SampleNum][i]->SetDirectory(nullptr);
456 BetaHist[SampleNum][i]->GetXaxis()->SetTitle("#beta parameter value");
457 BetaHist[SampleNum][i]->GetYaxis()->SetTitle("Counts");
458 }
459 } //end loop over bins
460 }//end loop over samples
461 }
462 first_pass = false;
463
464 // Loop over the samples
465 #ifdef MULTITHREAD
466 #pragma omp parallel for
467 #endif
468 for (int SampleNum = 0; SampleNum < nSamples; ++SampleNum)
469 {
470 if (SampleVector[SampleNum] == nullptr) continue;
471 if(doShapeOnly) NormaliseTH2Poly(SampleVector[SampleNum]);
472 // Loop over the distribution and fill the prior/posterior predictive
473 for (int i = 1; i <= maxBins[SampleNum]; ++i) {
474 const double Content = SampleVector[SampleNum]->GetBinContent(i);
475 PosteriorHist[SampleNum][i]->Fill(Content, Weight);
476 const double w2 = W2Vec[SampleNum]->GetBinContent(i);
477 w2Hist[SampleNum][i]->Fill(w2, Weight);
478 if(DoBetaParam)
479 {
480 const double data = DataHist[SampleNum]->GetBinContent(i);
481 const double BetaParam = GetBetaParameter(data, Content, w2, likelihood);
482 BetaHist[SampleNum][i]->Fill(BetaParam, Weight);
483 }
484 } // end bin loop
485 } // end samples loop
486} // end AddThrow

2409// ****************
2410// Get the mode error from a TH1D
2411double SampleSummary::GetModeError(TH1D* hpost){
2412// ****************
2413 // Get the bin which has the largest posterior density
2414 int MaxBin = hpost->GetMaximumBin();
2415
2416 // The total integral of the posterior
2417 const double Integral = hpost->Integral();
2418
2419 int LowBin = MaxBin;
2420 int HighBin = MaxBin;
2421 double sum = hpost->GetBinContent(MaxBin);;
2422 double LowCon = 0.0;
2423 double HighCon = 0.0;
2424 while (sum/Integral < 0.6827 && (LowBin > 0 || HighBin < hpost->GetNbinsX()+1) )
2425 {
2426 LowCon = 0.0;
2427 HighCon = 0.0;
2428 //KS:: Move further only if you haven't reached histogram end
2429 if(LowBin > 1)
2430 {
2431 LowBin--;
2432 LowCon = hpost->GetBinContent(LowBin);
2433 }
2434 if(HighBin < hpost->GetNbinsX())
2435 {
2436 HighBin++;
2437 HighCon = hpost->GetBinContent(HighBin);
2438 }
2439
2440 // If we're on the last slice and the lower contour is larger than the upper
2441 if ((sum+LowCon+HighCon)/Integral > 0.6827 && LowCon > HighCon) {
2442 sum += LowCon;
2443 break;
2444 // If we're on the last slice and the upper contour is larger than the lower
2445 } else if ((sum+LowCon+HighCon)/Integral > 0.6827 && HighCon >= LowCon) {
2446 sum += HighCon;
2447 break;
2448 } else {
2449 sum += LowCon + HighCon;
2450 }
2451 }
2452
2453 double Mode_Error = 0.0;
2454 if (LowCon > HighCon) {
2455 Mode_Error = std::fabs(hpost->GetBinCenter(LowBin)-hpost->GetBinCenter(MaxBin));
2456 } else {
2457 Mode_Error = std::fabs(hpost->GetBinCenter(HighBin)-hpost->GetBinCenter(MaxBin));
2458 }

2545// ****************
2546// Get the Watanabe-Akaike information criterion (WAIC)
2547void SampleSummary::StudyWAIC() {
2548// ****************
2549 // log pointwise predictive density
2550 double lppd = 0.;
2551 // effective number of parameters
2552 double p_WAIC = 0.;
2553
2554 #ifdef MULTITHREAD
2555 #pragma omp parallel for reduction(+:lppd, p_WAIC)
2556 #endif
2557 for (int SampleNum = 0; SampleNum < nSamples; ++SampleNum) {
2558 int nBins = maxBins[SampleNum];
2559 for (int i = 1; i <= nBins; ++i) {
2560 double mean_llh = 0.;
2561 double sum_exp_llh = 0;
2562 double mean_llh_squared = 0.;
2563
2564 for (unsigned int s = 0; s < nThrows; ++s) {
2565 const double data = DataHist[SampleNum]->GetBinContent(i);
2566 const double mc = MCVector[s][SampleNum]->GetBinContent(i);
2567 const double w2 = W2MCVector[s][SampleNum]->GetBinContent(i);
2568
2569 // Get the -log-likelihood for this sample and bin
2570 double neg_LLH_temp = SamplePDF->getTestStatLLH(data, mc, w2);
2571
2572 // Negate the negative log-likelihood to get the actual log-likelihood
2573 double LLH_temp = -neg_LLH_temp;
2574
2575 mean_llh += LLH_temp;
2576 mean_llh_squared += LLH_temp * LLH_temp;
2577 sum_exp_llh += std::exp(LLH_temp);
2578 }
2579
2580 // Compute the mean log-likelihood and the squared mean
2581 mean_llh /= nThrows;
2582 mean_llh_squared /= nThrows;
2583 sum_exp_llh /= nThrows;
2584 sum_exp_llh = std::log(sum_exp_llh);
2585
2586 // Log pointwise predictive density based on Eq. 4 in Gelman2014
2587 lppd += sum_exp_llh;
2588
2589 // Compute the effective number of parameters for WAIC
